-- Recipe 13: Map-Reduce Summarizer
-- Splits a large document into chunks, summarizes each chunk (map),
-- then combines all summaries into a final summary (reduce).
-- Demonstrates the map-reduce pattern for handling content beyond context limits.
--
-- Usage:
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama -m gemma3 document="$(cat large_doc.txt)"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="$(cat README.md)" style="bullet points"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="long text here..." style="executive brief"
--
-- Tools: chunk_plan(), extract_chunk() are in tools.py (CALL, zero tokens).
--        write_file(), read_file() are SPL built-ins — no --tools flag needed.
--        summarize_chunk(), reduce_summaries(), quality_score(), improve_summary() call the LLM.
--
-- Logs written to @log_dir (default: cookbook/13_map_reduce/logs):
--   chunk_00.md, chunk_01.md, ...    — raw extracted chunks
--   summary_00.md, summary_01.md, ... — per-chunk LLM summaries
--   final_summary.md                  — combined final summary

WORKFLOW map_reduce_summarizer
    INPUT:  
        @document TEXT, 
        @style TEXT,
        @log_dir TEXT DEFAULT 'cookbook/13_map_reduce/logs'
    OUTPUT: @final_summary TEXT
DO
    @chunk_index := 0
    @summaries := []          -- LIST: accumulates per-chunk summaries

    LOGGING 'Starting map-reduce | document length: ' + @document

    -- Step 1: Determine how to split the document (deterministic, 0 tokens)
    CALL chunk_plan(@document) INTO @chunk_count

    LOGGING 'Document split into ' + @chunk_count + ' chunks'

    -- Step 2: MAP — summarize each chunk independently
    WHILE @chunk_index < @chunk_count DO
        CALL extract_chunk(@document, @chunk_index, @chunk_count) INTO @chunk   -- deterministic, 0 tokens

        -- Save raw chunk to logs/
        CALL write_file(f'{@log_dir}/chunk_{@chunk_index}.md', @chunk) INTO @_

        GENERATE summarize_chunk(@chunk, @chunk_index) INTO @chunk_summary

        -- Save chunk summary to logs/
        CALL write_file(f'{@log_dir}/summary_{@chunk_index}.md', @chunk_summary) INTO @_

        LOGGING f'[Chunk {@chunk_index}/{@chunk_count}] summary saved'

        @summaries := list_append(@summaries, @chunk_summary)
        @chunk_index := @chunk_index + 1
    END

    -- Step 3: REDUCE — concat all chunk summaries, then combine into final
    @summaries_text := list_concat(@summaries, '\n')
    GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary

    -- Step 4: Quality check the final summary
    GENERATE quality_score(@final_summary, @document) INTO @score

    EVALUATE @score
        WHEN > 0.7 THEN
            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Final summary saved to {@log_dir}/final_summary.md (score={@score})'
            COMMIT @final_summary WITH status = 'complete', chunks = @chunk_count
        ELSE
            GENERATE improve_summary(@final_summary, @summaries_text) INTO @final_summary
            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Improved summary saved to {@log_dir}/final_summary.md (score={@score})'
            COMMIT @final_summary WITH status = 'refined', chunks = @chunk_count
    END

EXCEPTION
    WHEN ContextLengthExceeded THEN
        @summaries_text := list_concat(@summaries, '\n')
        GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary
        CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
        COMMIT @final_summary WITH status = 'partial'
    WHEN BudgetExceeded THEN
        @summaries_text := list_concat(@summaries, '\n')
        GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary
        CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
        COMMIT @final_summary WITH status = 'budget_limit'
END

from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer
import nltk
from sklearn.feature_extraction.text import TfidfVectorizer
from collections import defaultdict

nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')

def map_phase(text):
    """Tokenize the input text into individual words."""
    words = word_tokenize(text.lower())
    
    # Remove stop words and punctuation
    stop_words = set(stopwords.words('english'))
    words = [word for word in words if word.isalpha() and word not in stop_words]
    
    return [word for word in set(words) if word.isalpha()]

def lemmatize_text(text):
    """Lemmatize the input text."""
    lemmatizer = WordNetLemmatizer()
    words = word_tokenize(text.lower())
    lemmatized_words = [lemmatizer.lemmatize(word) for word in words]
    return ' '.join(lemmatized_words)

def reduce_phase(word_frequencies, top_n=10):
    """Sum up the TF-IDF scores for each unique word."""
    summary_words = []
    total_scores = defaultdict(int)
    
    for word, score in word_frequencies.items():
        total_scores[word] += score
        
    # Sort words by their sum of scores in descending order
    sorted_word_frequencies = sorted(total_scores.items(), key=lambda x: x[1], reverse=True)
    
    # Select the top N words with the highest scores
    for word, score in sorted_word_frequencies[:top_n]:
        summary_words.append(word)
        
    return ' '.join(summary_words)

def main():
    text1 = "The quick brown fox jumps over the lazy dog."
    print("Input 1 Summary: ", reduce_phase({word: tfidf_score for word, tfidf_score in map_phase(text1).items()}))

    # Input 2
    try:
        input_num = int(input())
        if input_num == 0:
            pass
    except ValueError:
        print("Invalid input. Please enter an integer.")
    
    text2 = "The quick brown fox jumps over the lazy dog. This is a test document for map-reduce summarization."
    text2 = lemmatize_text(text2)
    print("Input 2 Summary: ", reduce_phase({word: tfidf_score for word, tfidf_score in map_phase(text2).items()}))

if __name__ == "__main__":
    main()
