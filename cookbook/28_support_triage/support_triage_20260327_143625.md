CREATE FUNCTION support_categories()
RETURNS TEXT
AS $$
Ticket categories and their routing:
- billing:      finance team, SLA 4h,  high priority
                Covers: duplicate charges, wrong amounts, refund requests, invoice issues
- account:      auth team,    SLA 2h,  high priority
                Covers: login failures, password reset, account locked, 2FA issues
- shipping:     fulfillment,  SLA 6h,  high priority
                Covers: delayed delivery, wrong item received, missing package, tracking problems
- technical:    engineering,  SLA 8h,  medium priority
                Covers: app bugs, API errors, integration failures, data export issues
- product:      product team, SLA 24h, medium priority
                Covers: feature requests, how-to questions, documentation gaps
- complaint:    customer success, SLA 1h, urgent
                Covers: angry customers, escalation threats, repeated unresolved issues
- general:      first-line,   SLA 48h, low priority
                Covers: general inquiries, account queries, other
$$;

CREATE FUNCTION response_tone_guide(tone TEXT DEFAULT 'professional')
RETURNS TEXT
AS $$
SELECT CASE tone
  WHEN 'empathetic' THEN 'Warm and understanding. Acknowledge frustration. Use "I understand" and "I am sorry". Avoid jargon. Refer to the customer by first name.'
  WHEN 'formal'     THEN 'Professional and precise. Use formal language. Reference ticket numbers and order IDs explicitly. State next steps clearly.'
  WHEN 'friendly'   THEN 'Conversational and positive. Use the customer first name. End with an enthusiastic offer to help further.'
  ELSE                   'Professional but approachable. Concise. Action-oriented. Reference specific order details when available. No excessive apologies.'
END
$$;

CREATE FUNCTION order_context_prompt(order_context TEXT)
RETURNS TEXT
AS $$
SELECT CASE
  WHEN order_context = 'No order number found in this ticket.'
    THEN 'No order number was referenced — respond based on the ticket text alone.'
  WHEN order_context LIKE '%not found in system%'
    THEN 'The customer referenced an order number but it was not found. Ask them to confirm the order ID.'
  ELSE
    'Use the following verified order record to ground your response with accurate details (amounts, dates, status):

' || order_context
END
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW support_triage
    INPUT:  @ticket  TEXT,
            @product TEXT DEFAULT 'CloudDash',
            @tone    TEXT DEFAULT 'professional'
    OUTPUT: @drafted_response TEXT
DO
    LOGGING f'Support triage | product={@product} tone={@tone}' LEVEL INFO

    -- Step 1: Extract any order numbers mentioned in the ticket (deterministic tool — no LLM)
    CALL extract_order_numbers(@ticket) INTO @order_numbers

    -- Step 2: Fetch the matching order record(s) from orders.json (deterministic tool — no LLM)
    CALL lookup_order(@order_numbers) INTO @order_context
    LOGGING f'Order context loaded | order_numbers={@order_numbers}' LEVEL DEBUG

    -- Step 3: Classify the ticket using categories + order data as grounding context
    GENERATE classify_ticket(@ticket, support_categories(), order_context_prompt(@order_context)) INTO @classification_json
    LOGGING f'Classification: {@classification_json}' LEVEL DEBUG

    -- Step 4: Extract key structured details from the ticket
    GENERATE extract_ticket_details(@ticket, @order_context) INTO @details_json

    -- Step 5: Score urgency 0-10 based on ticket text and classification
    GENERATE detect_urgency(@ticket, @classification_json) INTO @urgency_score
    LOGGING f'Urgency score: {@urgency_score}' LEVEL INFO

    -- Step 6: Route — escalate immediately if urgency > 8, otherwise draft response
    EVALUATE @urgency_score
        WHEN > 8 THEN
            LOGGING f'High urgency — escalating | score={@urgency_score}' LEVEL WARN
            GENERATE escalation_alert(@ticket, @classification_json, @order_context) INTO @alert
            COMMIT @alert WITH status = 'escalated', priority = 'critical'
        ELSE
            -- Step 7: Draft a grounded, tone-appropriate response
            GENERATE draft_response(
                @ticket,
                @classification_json,
                @details_json,
                @order_context,
                @product,
                response_tone_guide(@tone)
            ) INTO @drafted_response

            -- Step 8: Quality-check the draft (0-10); revise once if score < 6
            GENERATE check_response_quality(@drafted_response, @ticket, @order_context) INTO @quality_score
            LOGGING f'Draft quality score: {@quality_score}' LEVEL DEBUG

            EVALUATE @quality_score
                WHEN < 6 THEN
                    LOGGING f'Quality below threshold — revising | score={@quality_score}' LEVEL WARN
                    GENERATE improve_response(@drafted_response, @ticket, @order_context) INTO @drafted_response
                    COMMIT @drafted_response WITH status = 'drafted_revised'
                ELSE
                    COMMIT @drafted_response WITH status = 'drafted', quality = @quality_score
            END
    END

EXCEPTION
    WHEN GenerationError THEN
        GENERATE fallback_response(@ticket, @product) INTO @drafted_response
        COMMIT @drafted_response WITH status = 'fallback'
END

import nltk
from nltk.sentiment import SentimentIntensityAnalyzer

# Initialize NLTK data needed for sentiment analysis
nltk.download('vader_lexicon')

class UrgencyDetector:
    def __init__(self):
        self.sia = SentimentIntensityAnalyzer()

    def detect_urgency(self, input_text):
        # Analyze the sentiment of the input text
        sentiment_scores = self.sia.polarity_scores(input_text)

        # Calculate the urgency score based on sentiment and keywords
        urgency_score = 0
        if 'error' in input_text.lower() or 'issue' in input_text.lower():
            urgency_score += 1.5
        elif 'charge' in input_text.lower() or 'duplicate' in input_text.lower():
            urgency_score += 2

        # Adjust the score based on sentiment intensity
        if sentiment_scores['compound'] < -0.5:
            urgency_score -= 0.5
        elif sentiment_scores['compound'] > 0.5:
            urgency_score += 1

        # Ensure the score is within the range [0, 10]
        urgency_score = max(0, min(urgency_score, 10))

        return {
            'classification': self._classify_urgency(urgency_score),
            'reasoning': self._provide_reasoning(input_text)
        }

    def _classify_urgency(self, urgency_score):
        if urgency_score > 7:
            return "Urgent"
        elif urgency_score > 4:
            return "High Priority"
        else:
            return "Low Priority"

    def _provide_reasoning(self, input_text):
        # Provide a short reasoning statement based on the input text
        if 'error' in input_text.lower() or 'issue' in input_text.lower():
            return "A clear error has been detected."
        elif 'charge' in input_text.lower() or 'duplicate' in input_text.lower():
            return "Duplicate charge detected, which may indicate unauthorized activity."
        else:
            return "No urgent issue detected."
