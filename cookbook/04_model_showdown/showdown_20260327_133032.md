```spl
-- Recipe 04: Model Showdown
-- Same prompt to multiple Ollama models — compare responses side-by-side.
-- Uses parallel CTEs instead of a bash loop for a pure-SPL implementation.
--
-- Usage:
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="What is the meaning of life?"
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="Explain recursion in 3 sentences" \
--       model_1=gemma3 model_2=llama3.2 model_3=mistral

CREATE FUNCTION compare_responses(
    prompt TEXT,
    model_1 TEXT, answer_1 TEXT,
    model_2 TEXT, answer_2 TEXT,
    model_3 TEXT, answer_3 TEXT
) RETURNS TEXT
AS $$
You are a neutral evaluator comparing responses from three AI models.

Prompt given to all models: "{prompt}"

First, reproduce each model's raw response exactly as given, formatted like this:

=== {model_1} ===
{answer_1}

=== {model_2} ===
{answer_2}

=== {model_3} ===
{answer_3}

Then provide your evaluation. For each model briefly assess:
- Response quality
- Key strengths or weaknesses

Conclude with which model gave the most helpful answer and why.
$$;

WORKFLOW model_showdown
    INPUT:
        @prompt  TEXT DEFAULT 'What is the meaning of life?',
        @model_1 TEXT DEFAULT 'gemma3',
        @model_2 TEXT DEFAULT 'phi4',
        @model_3 TEXT DEFAULT 'qwen2.5'
    OUTPUT: @comparison TEXT
DO
    -- Run the same prompt against all three models in parallel CTEs
    WITH
        response_1 AS (
            PROMPT ask_model_1
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_1
        ),
        response_2 AS (
            PROMPT ask_model_2
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_2
        ),
        response_3 AS (
            PROMPT ask_model_3
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_3
        )
    SELECT
        response_1.answer AS answer_1,
        response_2.answer AS answer_2,
        response_3.answer AS answer_3
    INTO @answer_1, @answer_2, @answer_3

    -- Judge all three responses together
    GENERATE compare_responses(
        @prompt,
        @model_1, @answer_1,
        @model_2, @answer_2,
        @model_3, @answer_3
    ) INTO @comparison

    COMMIT @comparison WITH
        status   = 'complete',
        model_1  = @model_1,
        model_2  = @model_2,
        model_3  = @model_3

EXCEPTION
    WHEN GenerationError THEN
        COMMIT 'One or more models failed to respond.' WITH status = 'error'
END
```
