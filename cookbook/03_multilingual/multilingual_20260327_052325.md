    system_role('You are a friendly assistant. Respond in the language specified.'),
    context.user_input AS input,
    context.lang AS lang

def greeting(name, lang='zh'):
    """
    Chinese greeting function
    
    Parameters:
    name (str): Name of the person to greet.
    lang (str): Language of greeting. Default is 'zh' for Chinese.
    
    Returns:
    str: The greeting message in the specified language.
    """
    if lang == 'zh':
        return f"亲爱的{name},你好!"
    elif lang == 'en':
        return f"Hello {name}!"
    else:
        raise ValueError("Unsupported language. Please use 'zh' for Chinese or 'en' for English.")

# Example usage
print(greeting("Wen-Guang", "zh"))  # Output: 亲爱的Wen-Guang，你好!
print(greeting("John"))  # Output: Hello John!
