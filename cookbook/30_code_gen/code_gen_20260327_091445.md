-- Generate a function, then generate its unit tests — in a single workflow.
-- Demonstrates sequential generation where output of step N is input to step N+1.
--
-- Usage:
--   spl run cookbook/30_code_gen/code_gen.spl --adapter ollama -m gemma3 \
--       spec="A function that validates an email address" language="Python"
--
--   spl run cookbook/30_code_gen/code_gen.spl --adapter ollama \
--       spec="Binary search over a sorted list, return index or -1" language="Go" \
--       test_framework="testing"

CREATE FUNCTION language_conventions(language TEXT)
RETURNS TEXT
AS $$
SELECT CASE language
  WHEN 'Python' THEN 'PEP 8 style. Type hints required. Docstring in Google format. No bare except.'
  WHEN 'Go'     THEN 'Standard Go conventions. Return (value, error). Table-driven tests. No panics in library code.'
  WHEN 'TypeScript' THEN 'Strict mode. Explicit return types. JSDoc comments. No any type.'
  WHEN 'Rust'   THEN 'Idiomatic Rust. Result<T, E> for fallible ops. No unwrap() in library code. Doc comments.'
  ELSE               'Clean, idiomatic code. Clear naming. Minimal dependencies. Well-commented edge cases.'
END
$$;

CREATE FUNCTION test_framework_guide(language TEXT, framework TEXT DEFAULT 'default')
RETURNS TEXT
AS $$
SELECT CASE language || ':' || framework
  WHEN 'Python:pytest'   THEN 'pytest. Use fixtures. Parametrize for multiple inputs. Assert with descriptive messages.'
  WHEN 'Python:unittest' THEN 'unittest.TestCase. setUp/tearDown. assertEqual, assertRaises. One assertion per test.'
  WHEN 'Go:testing'      THEN 'testing package. Table-driven tests with t.Run. t.Errorf not t.Fatalf unless truly fatal.'
  WHEN 'TypeScript:jest' THEN 'Jest. describe/it blocks. Mocks via jest.fn(). Async tests with async/await.'
  ELSE                        'Standard test framework for ' || language || '. Cover: happy path, edge cases, error cases.'
END
$$;

WORKFLOW code_gen_with_tests
    INPUT:
        @spec           TEXT,
        @language       TEXT DEFAULT 'Python',
        @test_framework TEXT DEFAULT 'default'
    OUTPUT: @final_output TEXT
DO
    LOGGING f'Code gen | language={@language} framework={@test_framework}' LEVEL INFO

    -- Step 1: Generate the implementation
    GENERATE implement_function(
        @spec,
        @language,
        language_conventions(@language)
    ) INTO @implementation
    LOGGING 'Implementation generated' LEVEL DEBUG

    -- Step 2: Self-review the implementation
    GENERATE review_implementation(@implementation, @spec, @language) INTO @review_notes

    -- Step 3: Refine if issues found
    EVALUATE @review_notes
        WHEN contains('issue') OR contains('error') OR contains('problem') THEN
            LOGGING 'Issues found — fixing implementation' LEVEL WARN
            GENERATE fix_implementation(@implementation, @review_notes, @language) INTO @implementation
        ELSE
            -- Implementation looks good
    END

    -- Step 4: Generate unit tests targeting the implementation
    LOGGING 'Generating unit tests ...' LEVEL DEBUG
    GENERATE generate_tests(
        @implementation,
        @spec,
        @language,
        test_framework_guide(@language, @test_framework)
    ) INTO @tests

    -- Step 5: Verify tests compile/parse (static check)
    GENERATE verify_test_syntax(@tests, @language) INTO @syntax_ok
    LOGGING f'Syntax check: {@syntax_ok}' LEVEL DEBUG

    -- Step 6: Assemble final output
    GENERATE assemble_output(@implementation, @tests, @language) INTO @final_output

    LOGGING f'Code gen complete | language={@language}' LEVEL INFO
    COMMIT @final_output WITH
        status = 'complete',
        language = @language,
        test_framework = @test_framework

EXCEPTION
    WHEN GenerationError THEN
        -- Return implementation without tests
        COMMIT @implementation WITH status = 'implementation_only', reason = 'test_generation_failed'
END
