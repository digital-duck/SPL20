-- Recipe 36: Tool-Use / Function-Call
-- Demonstrates registering Python functions as CALL-able tools in SPL 2.0.
--
-- Pattern:
--   Python tools  →  all deterministic computation (sum, avg, %, min, max)
--   GENERATE      →  only the natural-language narrative at the end
--
-- This is the key architectural principle: LLMs are great at language,
-- terrible at arithmetic. Keep them in their lane.
--
-- Usage:
--   spl run cookbook/36_tool_use/tool_use.spl \
--       --adapter ollama \
--       --tools cookbook/36_tool_use/tools.py \
--       sales="1200,1450,1380,1600,1750,1900" \
--       prev_total="7800" \
--       period="H1 2025"
--
-- Tool dispatch priority in CALL:
--   1. Python tool (registered via --tools)   ← used here for all math
--   2. SPL PROCEDURE
--   3. LLM fallback (last resort — avoid)

WORKFLOW sales_analysis
  INPUT:
    @sales     TEXT DEFAULT '1200,1450,1380,1600,1750,1900',
    @prev_total TEXT DEFAULT '7800',
    @period    TEXT DEFAULT 'H1 2025'
  OUTPUT: @report TEXT
DO
  -- All math delegated to Python tools — deterministic, instant, zero tokens
  CALL sum_values(@sales)            INTO @total
  CALL average_values(@sales)        INTO @avg
  CALL min_value(@sales)             INTO @low
  CALL max_value(@sales)             INTO @high
  CALL percentage_change(@prev_total, @total) INTO @growth
  CALL format_currency(@total)       INTO @total_fmt
  CALL format_currency(@avg)         INTO @avg_fmt

  -- Only the narrative goes to the LLM
  GENERATE sales_report(@period, @sales, @total_fmt, @avg_fmt, @low, @high, @growth) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN OTHERS THEN
    GENERATE error_summary(@period) INTO @report
    COMMIT @report WITH status = 'error'
END

def generate_sales_report(month, sales_data, revenue1, revenue2, cost1, cost2):
    """
    Generates a sales report for a given month and sales data.
    
    Parameters:
    month (str): The month of the sales report.
    sales_data (list): A list of sales figures.
    revenue1 (float): The first revenue figure.
    revenue2 (float): The second revenue figure.
    cost1 (float): The first cost figure.
    cost2 (float): The second cost figure.
    
    Returns:
    str: A formatted sales report.
    """
    
    # Format the month
    formatted_month = f"H{month} {2025}"
    
    # Calculate total sales and costs
    total_sales = sum(sales_data)
    total_costs = cost1 + cost2
    
    # Calculate gross profit and net income
    gross_profit = revenue1 - total_costs
    net_income = gross_profit - revenue2
    
    # Format the sales data for output
    formatted_sales_data = ", ".join(str(x) for x in sales_data)
    
    # Format the revenues and costs for output
    formatted_revenue1 = f"USD {revenue1:,.2f}"
    formatted_cost1 = f"USD {cost1:,.2f}"
    formatted_revenue2 = f"USD {revenue2:,.2f}"
    formatted_cost2 = f"USD {cost2:,.2f}"
    
    # Format the net income for output
    formatted_net_income = f"USD {net_income:,.2f}"
    
    # Determine if there's an increase in sales
    if gross_profit > 0 and total_sales > sales_data[0]:
        increase_percentage = ((total_sales - sales_data[0]) / sales_data[0]) * 100
        formatted_increase = f"+{increase_percentage:.2f}%"
    else:
        formatted_increase = "No increase in sales"
    
    # Format the output string
    report = (
        f"Sales Report for {formatted_month}:\n"
        f"{formatted_sales_data}\n"
        f"Gross Profit: {formatted_revenue1} - {formatted_net_income}\n"
        f"Increase: {formatted_increase}\n"
    )
    
    return report
