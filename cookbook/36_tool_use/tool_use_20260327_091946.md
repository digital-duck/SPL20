-- Recipe 36: Tool-Use / Function-Call
-- Demonstrates registering Python functions as CALL-able tools in SPL 2.0.
--
-- Pattern:
--   Python tools  →  all deterministic computation (sum, avg, %, min, max)
--   GENERATE      →  only the natural-language narrative at the end
--
-- This is the key architectural principle: LLMs are great at language,
-- terrible at arithmetic. Keep them in their lane.
--
-- Usage:
--   spl run cookbook/36_tool_use/tool_use.spl \
--       --adapter ollama \
--       --tools cookbook/36_tool_use/tools.py \
--       sales="1200,1450,1380,1600,1750,1900" \
--       prev_total="7800" \
--       period="H1 2025"
--
-- Tool dispatch priority in CALL:
--   1. Python tool (registered via --tools)   ← used here for all math
--   2. SPL PROCEDURE
--   3. LLM fallback (last resort — avoid)

WORKFLOW sales_analysis
  INPUT:
    @sales     TEXT DEFAULT '1200,1450,1380,1600,1750,1900',
    @prev_total TEXT DEFAULT '7800',
    @period    TEXT DEFAULT 'H1 2025'
  OUTPUT: @report TEXT
DO
  -- All math delegated to Python tools — deterministic, instant, zero tokens
  CALL sum_values(@sales)            INTO @total
  CALL average_values(@sales)        INTO @avg
  CALL min_value(@sales)             INTO @low
  CALL max_value(@sales)             INTO @high
  CALL percentage_change(@prev_total, @total) INTO @growth
  CALL format_currency(@total)       INTO @total_fmt
  CALL format_currency(@avg)         INTO @avg_fmt

  -- Only the narrative goes to the LLM
  GENERATE sales_report(@period, @sales, @total_fmt, @avg_fmt, @low, @high, @growth) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN OTHERS THEN
    GENERATE error_summary(@period) INTO @report
    COMMIT @report WITH status = 'error'
END

class SalesReport:
    def __init__(self, year, data_points):
        self.year = year
        self.data_points = data_points
        self.total_sales = 0
        self.average_monthly_sales = 0
        self.net_profit = 0

    def calculate_total_sales(self):
        """Calculate the total sales based on the provided data points."""
        self.total_sales = sum(self.data_points)

    def calculate_average_monthly_sales(self):
        """Calculate the average monthly sales based on the provided data points."""
        if len(self.data_points) > 1:
            self.average_monthly_sales = self.total_sales / len(self.data_points)
        else:
            print("At least two data points are required to calculate the average monthly sales.")

    def calculate_net_profit(self):
        """Calculate the net profit based on the provided gross profit and total sales."""
        # Assuming we have a list of monthly gross profits
        monthly_gross_profits = [1200, 1450, 1380, 1600, 1750, 1900]
        
        # Calculate the difference between each month's revenue and cost
        monthly_net_sales = []
        for i in range(len(monthly_gross_profits)):
            net_sales = self.data_points[i] - monthly_gross_profits[i]
            if i > 0:
                previous_month_net_sales = monthly_net_sales[-1]
                net_sales += previous_month_net_sales
            monthly_net_sales.append(net_sales)

        # Calculate the total gross profit and net sales
        for month, month_sales in zip(monthly_gross_profits, self.data_points):
            if month > 0:
                net_profit = month - month_sales + monthly_net_sales[-1]
            else:
                net_profit = month

        self.net_profit = net_profit


def generate_report(year, data_points, gross_profit, total_revenue, percent_change):
    report = SalesReport(year, data_points)
    
    # Validate the input values
    if not all(isinstance(x, (int, float)) for x in data_points) or any(x < 0 for x in data_points):
        print("Invalid data points. All values should be positive numbers.")
        return
    
    report.calculate_total_sales()
    report.calculate_average_monthly_sales()
    report.calculate_net_profit()

    # Format the gross profit and total revenue
    gross_profit_str = "${:,.2f}".format(gross_profit)
    total_revenue_str = "${:,.2f}".format(total_revenue)

    # Calculate percent change
    if percent_change != 0:
        net_profit_percent_change = ((gross_profit - report.net_profit) / gross_profit) * 100
    else:
        net_profit_percent_change = 0

    return {
        "year": year,
        "total_sales": report.total_sales,
        "average_monthly_sales": report.average_monthly_sales,
        "net_profit": report.net_profit,
        "gross_profit": gross_profit_str,
        "total_revenue": total_revenue_str,
        "percent_change": f"{net_profit_percent_change:.2f}%",
    }
