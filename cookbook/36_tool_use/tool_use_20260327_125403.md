-- Recipe 36: Tool-Use / Function-Call
-- Demonstrates registering Python functions as CALL-able tools in SPL 2.0.
--
-- Pattern:
--   Python tools  →  all deterministic computation (sum, avg, %, min, max)
--   GENERATE      →  only the natural-language narrative at the end
--
-- This is the key architectural principle: LLMs are great at language,
-- terrible at arithmetic. Keep them in their lane.
--
-- Usage:
--   spl run cookbook/36_tool_use/tool_use.spl \
--       --adapter ollama \
--       --tools cookbook/36_tool_use/tools.py \
--       sales="1200,1450,1380,1600,1750,1900" \
--       prev_total="7800" \
--       period="H1 2025"
--
-- Tool dispatch priority in CALL:
--   1. Python tool (registered via --tools)   ← used here for all math
--   2. SPL PROCEDURE
--   3. LLM fallback (last resort — avoid)

WORKFLOW sales_analysis
  INPUT:
    @sales     TEXT DEFAULT '1200,1450,1380,1600,1750,1900',
    @prev_total TEXT DEFAULT '7800',
    @period    TEXT DEFAULT 'H1 2025'
  OUTPUT: @report TEXT
DO
  -- All math delegated to Python tools — deterministic, instant, zero tokens
  CALL sum_values(@sales)            INTO @total
  CALL average_values(@sales)        INTO @avg
  CALL min_value(@sales)             INTO @low
  CALL max_value(@sales)             INTO @high
  CALL percentage_change(@prev_total, @total) INTO @growth
  CALL format_currency(@total)       INTO @total_fmt
  CALL format_currency(@avg)         INTO @avg_fmt

  -- Only the narrative goes to the LLM
  GENERATE sales_report(@period, @sales, @total_fmt, @avg_fmt, @low, @high, @growth) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN OTHERS THEN
    GENERATE error_summary(@period) INTO @report
    COMMIT @report WITH status = 'error'
END

def process_sales_report(date, sales_data, revenue1, revenue2, unit_price1, unit_price2, increase):
    """
    Process sales report and calculate total sales and profit.

    Args:
        date (str): Date of the sales report.
        sales_data (list): List of daily sales data.
        revenue1 (float): Revenue from a specific source in USD.
        revenue2 (float): Revenue from another specific source in USD.
        unit_price1 (float): Unit price of product in USD.
        unit_price2 (float): Unit price of product in USD.
        increase (float): Percentage increase.

    Returns:
        tuple: Total sales and profit.
    """
    
    # Convert revenue data to float
    if not isinstance(revenue1, str):
        revenue1 = float(revenue1.replace('USD ', ''))
    if not isinstance(revenue2, str):
        revenue2 = float(revenue2.replace('USD ', ''))

    # Convert unit price data to float
    if not isinstance(unit_price1, str):
        unit_price1 = float(unit_price1.replace('USD ', ''))
    if not isinstance(unit_price2, str):
        unit_price2 = float(unit_price2.replace('USD ', ''))

    # Calculate total sales by summing up the daily sales data
    total_sales = sum(sales_data)

    # Calculate profit from revenue 1
    profit_revenue1 = revenue1 - (unit_price1 * total_sales)

    # Calculate profit from revenue 2
    profit_revenue2 = revenue2 - (unit_price2 * total_sales)

    # Apply percentage increase to sales and revenue
    increased_profit = profit_revenue1 + (profit_revenue1 * (increase / 100))

    return (
        date,
        f"Total Sales: {total_sales}",
        f"Profit from Revenue 1: {profit_revenue1:.2f} USD",
        f"Profit from Revenue 2: {profit_revenue2:.2f} USD",
        f"Increased Profit: {increased_profit:.2f} USD"
    )
