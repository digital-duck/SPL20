    INPUT: @topic TEXT
    OUTPUT: @summary TEXT
DO
    -- Step 1: Research
    GENERATE research(@topic) INTO @research

    -- Step 2: Analyze
    GENERATE analyze(@research) INTO @analysis

    -- Step 3: Summarize
    GENERATE summarize(@analysis) INTO @summary

    COMMIT @summary WITH status = 'complete'
END
