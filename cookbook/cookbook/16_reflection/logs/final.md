import uuid
from flask import Flask, request, jsonify
app = Flask(__name__)

# Define database schema
class UrlDatabase:
    def __init__(self):
        self.urls = {}
        self.url_mappings = {}

    # Create a new URL mapping
    def create_url_mapping(self, original_url_id, shortened_url_id):
        if original_url_id not in self.urls:
            raise ValueError("Original URL ID does not exist")
        if shortened_url_id not in self.url_mappings:
            raise ValueError("Shortened URL ID does not exist")

        self.url_mappings[original_url_id] = shortened_url_id

    # Get the original URL for a shortened URL
    def get_original_url(self, shortened_url):
        if shortened_url not in self.urls:
            return None

        original_url_id = self.urls[shortened_url]
        return original_url_id

# Create a new instance of the database
database = UrlDatabase()

class UrlShortener:
    def __init__(self, database):
        self.database = database

    # Shorten a URL and store it in the database
    def shorten_url(self, original_url):
        if not original_url:
            raise ValueError("Original URL cannot be empty")

        short_id = str(uuid.uuid4())
        shortened_url = f'http://example.com/{short_id}'
        self.database.urls[shortened_url] = original_url

        return {'shortened_url': shortened_url}

    # Get the original URL for a shortened URL
    def get_original_url(self, shortened_url):
        return self.database.get_original_url(shortened_url)

# Define API endpoints
@app.route('/shorten', methods=['POST'])
def shorten():
    original_url = request.json['original_url']
    url_shortener = UrlShortener(database)
    shortened_data = url_shortener.shorten_url(original_url)
    return jsonify(shortened_data)

@app.route('/get-original', methods=['GET'])
def get_original():
    shortened_url = request.args.get('shortened_url')
    if not shortened_url:
        raise ValueError("Shortened URL is required")

    original_url_id = url_shortener.get_original_url(shortened_url)
    return jsonify({'original_url': original_url_id})

@app.route('/update-mapping', methods=['POST'])
def update_mapping():
    original_url_id = request.json['original_url_id']
    shortened_url_id = request.json['shortened_url_id']

    if not original_url_id:
        raise ValueError("Original URL ID is required")
    if not shortened_url_id:
        raise ValueError("Shortened URL ID is required")

    url_shortener = UrlShortener(database)
    try:
        url_shortener.create_url_mapping(original_url_id, shortened_url_id)
        return jsonify({'message': 'URL mapping updated successfully'})
    except Exception as e:
        # Log error
        import logging
        logging.error('Error updating URL mapping: %s', e)
        raise ValueError("Failed to update URL mapping")

if __name__ == '__main__':
    app.run(debug=True)
