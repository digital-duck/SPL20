    def generate_question_set(self, senior_swe, available_roles, focus_area, difficulty_level, num_questions):
        question_sets = []
        
        # Add questions for the given focus area and role
        if senior_swe:
            questions = [
                f"What is your approach to designing a scalable system?",
                "How do you handle system complexity in your design?"
            ]
        else:
            questions = []

        # Randomly select questions from available roles based on difficulty level
        for role in available_roles:
            question = random.choice(role.focus_areas)
            if question.lower() == focus_area.lower():
                continue
            else:
                if role.name != senior_swe and difficulty_level.lower() == "medium":
                    questions.append(f"What are some key considerations when implementing {question} in a {role.name} project?")
        # Add more questions based on the selected difficulty level
        if difficulty_level.lower() == "easy":
            for _ in range(num_questions):
                questions.append(random.choice(["What is your experience with?", "Can you explain how to?"]))
        elif difficulty_level.lower() == "hard":
            for _ in range(num_questions * 2):  # Generate more hard-level questions
                questions.append(f"How do you handle {random.choice(['edge cases', 'ambiguities', 'optimization'])} in your design?")

                questions.append(random.choice(["Can you walk me through", "How would you approach"]))
        
        # Shuffle the list of questions to randomize the order
        random.shuffle(questions)
        question_sets.append(questions)
