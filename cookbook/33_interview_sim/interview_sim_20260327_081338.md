$$;

Return valid JSON only — no prose outside the JSON:

$$;

from typing import List

class QuestionSetGenerator:
    def __init__(self, role: str, roles: dict, topic: str, difficulty: str, num_questions: int):
        self.role = role
        self.roles = roles
        self.topic = topic
        self.difficulty = difficulty
        self.num_questions = num_questions

    def generate_question_set(self) -> List[dict]:
        question_set = []

        # Get the focus areas for the given role
        focus_areas = self.get_focus_areas()

        # Generate questions based on the focus areas and topic
        for area in focus_areas:
            if 'machine_learning' in area['focus_areas'] and self.topic == 'system_design':
                continue

            question = {
                'question': f"What is your experience with {area['name']}?",
                'answer': '',
                'difficulty': self.difficulty,
                'role': self.role
            }
            question_set.append(question)

        # Ensure the number of questions does not exceed the specified limit
        if len(question_set) > self.num_questions:
            question_set = question_set[:self.num_questions]

        return question_set

    def get_focus_areas(self) -> List[dict]:
        # Return a list of focus areas for the given role
        return [self.roles[role] for role in self.roles if role != self.role]
