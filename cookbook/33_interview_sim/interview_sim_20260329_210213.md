class EvaluationRubric:
    def __init__(self):
        self.evaluation_points = {
            "senior_swe": {"system_design": 10, "algorithms": 10, "behavioral": 10},
            "data_scientist": {"machine_learning": 10, "statistics": 10, "python_and_data_engineering": 10},
            "devops_engineer": {"kubernetes": 10, "cicd": 10, "cloud_infrastructure": 10}
        }

    def calculate_score(self, answers):
        total_points = 0
        for role, points in self.evaluation_points.items():
            if role in answers:
                for point_type in points:
                    point_value = points[point_type]
                    # Check the format of answer to assign score.
                    try:
                        total_points += point_value * float(answers[role][point_type])
                    except Exception as e:
                        print(f"Exception: {e}")
        return total_points

def score_answer(role, question):
    """
    Scores a candidate's answer for a specific role.

    Parameters:
    role (str): The role being evaluated.
    question (dict): A dictionary containing the question and its corresponding scoring criteria.

    Returns:
    int: The score assigned to the candidate's answer (out of 10).
    """

    # Map roles to their respective focus areas
    role_map = {
        "senior_swe": ["system_design", "algorithms"],
        "data_scientist": ["machine_learning", "statistics"],
        "devops_engineer": ["kubernetes", "cicd", "cloud_infrastructure"]
    }

    # Get the focus areas for the given role
    focus_areas = role_map.get(role, [])

    # Initialize score to 0
    score = 0

    # Check if the question type is valid
    if question["type"] not in ["candidate_skills", "company_culture"]:
        return "Invalid question type"

    # Get the scoring criteria for the given question type and role
    if question["type"] == "candidate_skills":
        skills = question.get("skills")
        behavioral_questions = question.get("behavioral_questions")
    elif question["type"] == "company_culture":
        culture_insights = question.get("culture_insights")

    # Score the candidate's answer based on their focus areas and the given question criteria
    if skills:
        for skill in skills:
            if skill in focus_areas:
                score += 1

        if behavioral_questions:
            for behavior, importance in behavioral_questions.items():
                score += importance * (skill in focus_areas)

    elif culture_insights:
        for insight, weight in culture_insights.items():
            score += weight * (insight in focus_areas)

    # Ensure the score does not exceed 10
    score = min(score, 10)

    return score
