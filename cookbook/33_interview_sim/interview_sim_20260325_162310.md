-- ── Persona functions (zero LLM cost — injected as system context) ──────────

CREATE FUNCTION interviewer_persona(role TEXT, focus TEXT, difficulty TEXT DEFAULT 'medium')
RETURNS TEXT AS $$
You are an experienced technical interviewer at a top tech company.
Role being interviewed for: {role}
Focus area for this session: {focus}
Difficulty level: {difficulty}

Rules:
- Ask targeted, probing questions. Follow up on shallow or vague answers.
- Be professional and neutral — neither warm nor hostile.
- Ask exactly one question at a time.
- Do not reveal what a good answer looks like before hearing the response.
- If the candidate's answer is incomplete, push deeper with a follow-up.
$$;

CREATE FUNCTION candidate_persona(role TEXT, experience TEXT DEFAULT '5 years')
RETURNS TEXT AS $$
You are a job candidate interviewing for: {role}
You have {experience} of relevant experience.

Rules:
- Answer honestly based on the candidate profile you have been given.
- Think out loud and be specific — reference real or plausible past experiences.
- Show genuine strengths confidently. Show gaps honestly without guessing wildly.
- If unsure, say so and explain your reasoning process rather than bluffing.
$$;

CREATE FUNCTION evaluation_rubric()
RETURNS TEXT AS $$
Evaluate the candidate answer on four dimensions (each 0-10):
1. accuracy     — Is the answer technically correct and free of errors?
2. depth        — Does the candidate go beyond the surface level?
3. communication — Is the answer clear, structured, and easy to follow?
4. experience   — Does the candidate reference concrete real-world scenarios?

Also derive: total = accuracy + depth + communication + experience  (max 40)

Return valid JSON only — no prose outside the JSON:
{"accuracy": N, "depth": N, "communication": N, "experience": N, "total": N, "feedback": "one concise sentence"}
$$;

WORKFLOW interview_sim

            @role          TEXT DEFAULT '',

    OUTPUT: @evaluation_report TEXT
DO

    CALL extract_question(@questions_json, '2') INTO @q2

    -- Step 5: Candidate answers each question (LLM, grounded by candidate profile)
    GENERATE answer_question(
        @q1, @role, @role_context, @candidate_profile,
        candidate_persona(@role, @experience)
    ) INTO @a1

    GENERATE answer_question(
        @q2, @role, @role_context, @candidate_profile,
        candidate_persona(@role, @experience)
    ) INTO @a2

    GENERATE answer_question(
        @q3, @role, @role_context, @candidate_profile,
        candidate_persona(@role, @experience)
    ) INTO @a3

    -- Step 6: Score each answer against the rubric (LLM)
    LOGGING 'Scoring answers ...' LEVEL DEBUG
    GENERATE score_answer(@q1, @a1, @role_context, evaluation_rubric()) INTO @score1
    GENERATE score_answer(@q2, @a2, @role_context, evaluation_rubric()) INTO @score2
    GENERATE score_answer(@q3, @a3, @role_context, evaluation_rubric()) INTO @score3

    -- Step 7: Aggregate scores across all questions (deterministic — no LLM)
    CALL aggregate_scores(@score1, @score2, @score3) INTO @agg_scores
    LOGGING f'Aggregate scores: {@agg_scores}' LEVEL INFO

    -- Step 8: Compile the interview transcript (deterministic — no LLM)
    CALL compile_transcript(
        @q1, @a1, @score1,
        @q2, @a2, @score2,
        @q3, @a3, @score3,
        @role, @focus
    ) INTO @transcript

    -- Step 9: Write the overall evaluation narrative (LLM, receives deterministic agg_scores)
    GENERATE overall_evaluation(
        @transcript,
        @agg_scores,
        @role,
        @focus,
        @candidate_profile
    ) INTO @evaluation_report

    LOGGING f'Evaluation complete | role={@role} focus={@focus}' LEVEL INFO
    COMMIT @evaluation_report WITH
        status     = 'complete',
        role       = @role,
        focus      = @focus,
        difficulty = @difficulty

EXCEPTION
    WHEN GenerationError THEN
        COMMIT @transcript WITH status = 'partial', reason = 'evaluation_failed'
END

```python
def score_answer(answer, question, role):
    """
    Scores a user's answer to a question based on their role and the question's evaluation rubric.

    Args:
        answer (str): The user's answer.
        question (dict): The question being asked, including its evaluation rubric.
        role (str): The user's role.

    Returns:
        int: A score out of 10 for the user's answer.
    """

    # Initialize the score to 0
    score = 0

    # Extract the relevant focus areas from the question and role
    relevant_focus_areas = set(question['evaluation_rubric']['focus_areas']) | {role}

    # Check if the user answered a question related to their focus area
    for answer_key, key in enumerate(question['answer_keys']):
        if any(answer_key.lower() in answer.lower() for answer_key in relevant_focus_areas):
            score += 10 - (question['evaluation_rubric']['weighting'][key] * 2)

    # Ensure the score is between 0 and 10
    score = max(0, min(score, 10))

    return int(score)
```
