-- Recipe 31: Sentiment Pipeline
-- Batch sentiment over a list of inputs, aggregate trend statistics.
-- Demonstrates iteration over a list with per-item structured output + aggregation,
-- with deterministic tool calls for loading, splitting, stats, and extreme detection.
--
-- Usage (load from file):
--   spl2 run cookbook/31_sentiment_pipeline/sentiment.spl \
--       --adapter ollama -m gemma3 \
--       --tools cookbook/31_sentiment_pipeline/tools.py \
--       filename=product_reviews.txt \
--       domain=product_reviews
--
-- Usage (inline pipe-delimited):
--   spl2 run cookbook/31_sentiment_pipeline/sentiment.spl \
--       --adapter ollama -m gemma3 \
--       --tools cookbook/31_sentiment_pipeline/tools.py \
--       items="Great product, love it! | Terrible experience, never again | It was okay I guess" \
--       delimiter="|" \
--       domain=product_reviews
--
-- Sample files in reviews/:
--   product_reviews.txt   15 product reviews (positive, negative, mixed)
--   support_tickets.txt   12 customer support messages
--   social_media.txt      15 social media posts

-- ── Schema (zero LLM cost — injected as grounding context) ─────────────────

CREATE FUNCTION sentiment_schema()
RETURNS JSON AS $$
{
  "type": "array",
  "description": "One entry per input item, in the same order as the input list.",
  "items": {
    "type": "object",
    "properties": {
      "item":       {"type": "string", "description": "The original text that was analysed"},
      "label":      {"type": "string", "enum": ["positive", "negative", "neutral", "mixed"]},
      "score":      {"type": "number", "minimum": -1, "maximum": 1,
                     "description": "Sentiment score: -1 most negative, +1 most positive"},
      "confidence": {"type": "number", "minimum": 0, "maximum": 1,
                     "description": "Model confidence in the label"},
      "emotions":   {"type": "array", "items": {"type": "string"},
                     "description": "Detected emotions e.g. joy, anger, frustration, surprise"},
      "key_phrases":{"type": "array", "items": {"type": "string"},
                     "description": "Most sentiment-bearing phrases from the text"}
    },
    "required": ["item", "label", "score", "confidence"]
  }
}
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW sentiment_pipeline
    INPUT:  @filename  TEXT DEFAULT '',
            @items     TEXT DEFAULT '',
            @delimiter TEXT DEFAULT '\n',
            @domain    TEXT DEFAULT 'general'
    OUTPUT: @report TEXT
DO
    -- Step 1: Load items from file if filename provided (deterministic — no LLM)
    CALL load_items(@filename) INTO @file_content

    -- Step 2: Split into a clean JSON array (deterministic — no LLM)
    CALL split_items(@file_content, @items, @delimiter) INTO @item_list

    -- Step 3: Batch sentiment — one structured result per item (LLM)
    GENERATE batch_sentiment(@item_list, sentiment_schema(), @domain) INTO @sentiment_results

    -- Step 4: Compute aggregate statistics (deterministic — no LLM)
    CALL compute_stats(@sentiment_results) INTO @stats_json

    -- Step 5: Identify extreme and notable items (deterministic — no LLM)
    CALL find_extremes(@sentiment_results) INTO @extremes

    -- Step 6: Generate trend narrative from stats and extremes (LLM)
    GENERATE summarize_sentiment_trends(
        @stats_json,
        @extremes,
        @domain
    ) INTO @trend_summary

    -- Step 7: Assemble the full report (LLM)
    GENERATE assemble_sentiment_report(
        @sentiment_results,
        @stats_json,
        @trend_summary,
        @extremes,
        @domain
    ) INTO @report

    COMMIT @report WITH status = 'complete', domain = @domain

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Too many items — skip per-item detail, return stats + trend only
        CALL compute_stats(@sentiment_results) INTO @stats_json
        CALL find_extremes(@sentiment_results) INTO @extremes
        GENERATE summarize_sentiment_trends(@stats_json, @extremes, @domain) INTO @trend_summary
        SET @report = @trend_summary
        COMMIT @report WITH status = 'stats_only', reason = 'context_length_exceeded'
END

def batch_sentiment(texts):

    X = vectorizer.transform(texts)
    
    # Calculate the sentiment scores for each text using the trained classifier
    predictions = clf.predict_proba(X)
    return predictions

def assemble_sentiment_report(texts, threshold=0.5):
    sentiments = []
    for text in texts:
        prediction = batch_sentiment([text])
        sentiment = 'Positive' if prediction[0][1] > threshold else 'Negative'
        sentiments.append((sentiment, 1))
    
    return sentiments
