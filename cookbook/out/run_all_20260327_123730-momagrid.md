-- Recipe 04: Model Showdown
-- Same prompt to multiple Ollama models — compare responses side-by-side.
-- Uses parallel CTEs instead of a bash loop for a pure-SPL implementation.
--
-- Usage:
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="What is the meaning of life?"
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="Explain recursion in 3 sentences" \
--       model_1=gemma3 model_2=llama3.2 model_3=mistral

CREATE FUNCTION compare_responses(
    prompt TEXT,
    model_1 TEXT, answer_1 TEXT,
    model_2 TEXT, answer_2 TEXT,
    model_3 TEXT, answer_3 TEXT
) RETURNS TEXT
AS $$
You are a neutral evaluator comparing responses from three AI models.

Prompt given to all models: "{prompt}"

First, reproduce each model's raw response exactly as given, formatted like this:

=== {model_1} ===
{answer_1}

=== {model_2} ===
{answer_2}

=== {model_3} ===
{answer_3}

Then provide your evaluation. For each model briefly assess:
- Response quality
- Key strengths or weaknesses

Conclude with which model gave the most helpful answer and why.
$$;

WORKFLOW model_showdown
    INPUT:
        @prompt  TEXT DEFAULT 'What is the meaning of life?',
        @model_1 TEXT DEFAULT 'gemma3',
        @model_2 TEXT DEFAULT 'phi4',
        @model_3 TEXT DEFAULT 'qwen2.5'
    OUTPUT: @comparison TEXT
DO
    -- Run the same prompt against all three models in parallel CTEs
    WITH
        response_1 AS (
            PROMPT ask_model_1
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_1
        ),
        response_2 AS (
            PROMPT ask_model_2
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_2
        ),
        response_3 AS (
            PROMPT ask_model_3
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_3
        )
    SELECT
        response_1.answer AS answer_1,
        response_2.answer AS answer_2,
        response_3.answer AS answer_3
    INTO @answer_1, @answer_2, @answer_3

    -- Judge all three responses together
    GENERATE compare_responses(
        @prompt,
        @model_1, @answer_1,
        @model_2, @answer_2,
        @model_3, @answer_3
    ) INTO @comparison

    COMMIT @comparison WITH
        status   = 'complete',
        model_1  = @model_1,
        model_2  = @model_2,
        model_3  = @model_3

EXCEPTION
    WHEN GenerationError THEN
        COMMIT 'One or more models failed to respond.' WITH status = 'error'
END

-- SPL 2.0: Self-Refine Pattern
-- Iteratively improves output through critique and refinement

WORKFLOW self_refine
  INPUT:
    @task TEXT,
    @max_iterations INTEGER DEFAULT 5,
    @log_dir TEXT DEFAULT 'cookbook/05_self_refine/logs'
  OUTPUT: @result TEXT
DO
  @iteration := 0

  LOGGING f'Self-refine started | max_iterations={@max_iterations}' LEVEL INFO

  -- Initial draft
  GENERATE draft(@task) INTO @current
  LOGGING 'Initial draft ready' LEVEL INFO
  CALL write_file(f'{@log_dir}/draft_0.md', @current) INTO @_

  -- Iterative refinement loop
  WHILE @iteration < @max_iterations DO
    LOGGING f'Iteration {@iteration} | critiquing ...' LEVEL DEBUG
    GENERATE critique(@current) INTO @feedback
    CALL write_file(f'{@log_dir}/feedback_{@iteration}.md', @feedback) INTO @_

    EVALUATE @feedback
      WHEN 'satisfactory' THEN
        LOGGING f'Satisfactory at iteration {@iteration}' LEVEL INFO
        CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
        COMMIT @current WITH status = 'complete', iterations = @iteration
      ELSE
        GENERATE refined(@current, @feedback) INTO @current
        @iteration := @iteration + 1
        CALL write_file(f'{@log_dir}/draft_{@iteration}.md', @current) INTO @_
        LOGGING f'Refined | iteration={@iteration}' LEVEL DEBUG
    END
  END

  -- If loop exhausted, commit best effort
  LOGGING f'Max iterations reached | iterations={@iteration}' LEVEL WARN
  CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
  COMMIT @current WITH status = 'max_iterations', iterations = @iteration

EXCEPTION
  WHEN MaxIterationsReached THEN
    CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
    COMMIT @current WITH status = 'partial'
  WHEN BudgetExceeded THEN
    COMMIT @current WITH status = 'budget_limit'
END

-- SPL 2.0: ReAct Agent — Population Growth
-- Fetch a country's population for the past two years via WebSearch,
-- then compute year-over-year growth rate using a Python tool (no LLM math).
--
-- Usage:
--   spl run cookbook/06_react_agent/react_agent.spl \
--       --adapter claude_cli -m claude-sonnet-4-6 \
--       --allowed-tools WebSearch \
--       --tools cookbook/06_react_agent/tools.py \
--       country="France"

-- Tight prompt for search_population: return only the number, no narration.
CREATE FUNCTION search_population(country TEXT, year INT) RETURNS TEXT 
AS $$
Use WebSearch to find the total population of {country} for the year {year}.

Return ONLY the population as a plain integer with no commas, spaces, or text.
Example of valid output: 1409670000
No explanation. No units. No markdown. Just the number.
$$;

-- SPL PROCEDURE: search the web for a country's population in a specific year.
PROCEDURE search_population(country TEXT, year INT)
RETURNS TEXT
DO
  GENERATE search_population(country, year) INTO @result
  COMMIT @result
END

-- calc_growth_rate is a Python tool registered from tools.py via --tools.
-- CALL routes to Python directly — deterministic, zero LLM cost.

-- Tight prompt for growth_report: plain narrative, no tool use, no file writes.
CREATE FUNCTION growth_report(country TEXT, year_prev TEXT, pop_prev TEXT, year_curr TEXT, pop_curr TEXT, growth_rate TEXT) 
RETURNS TEXT 
AS $$
Write a concise 2-3 sentence population growth report for {country}.

Data:
- {year_prev} population: {pop_prev}
- {year_curr} population: {pop_curr}
- Year-over-year growth rate: {growth_rate}%

Plain prose only. No markdown headers. No file creation. No tool use. Just the narrative.
$$;

WORKFLOW population_growth
  INPUT:
    @country TEXT DEFAULT 'China',
    @year_curr INTEGER DEFAULT 2023
  OUTPUT: @report TEXT
DO
  @year_prev := @year_curr - 1

  LOGGING f'Population growth | country={@country} years={@year_prev}-{@year_curr}' LEVEL INFO

  -- Step 1: Fetch population for each year via WebSearch
  CALL search_population(@country, @year_prev) INTO @pop_prev
  CALL search_population(@country, @year_curr) INTO @pop_curr
  LOGGING f'Population fetched | {@year_prev}={@pop_prev} {@year_curr}={@pop_curr}' LEVEL DEBUG

  -- Step 2: Compute growth rate — routed to Python tool, not LLM
  CALL calc_growth_rate(@pop_prev, @pop_curr) INTO @growth_rate
  LOGGING f'Growth rate computed: {@growth_rate}%' LEVEL INFO

  -- Step 3: Compose the final human-readable report
  GENERATE growth_report(@country, @year_prev, @pop_prev, @year_curr, @pop_curr, @growth_rate) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN SearchFailed THEN
    GENERATE search_error_report(@country) INTO @report
    COMMIT @report WITH status = 'error'
END

-- Automated testing of key cookbook recipes across multiple models.
-- Uses parallel CTEs to fan out all recipe × model combinations at once,
-- then generates a structured PASS/FAIL report.
--
-- The bash script (batch_test.sh) is kept for CI/CD and shell-based pipelines.
--

    model_1 TEXT,
    model_2 TEXT,
    r1_name TEXT, r1_m1 TEXT, r1_m2 TEXT,
    r2_name TEXT, r2_m1 TEXT, r2_m2 TEXT,
    r3_name TEXT, r3_m1 TEXT, r3_m2 TEXT
) RETURNS TEXT

You are a test reporter. Review the outputs below and produce a concise pass/fail report.

A result is PASS if the output is a non-empty, coherent response.
A result is FAIL if the output is empty, contains an error, or is clearly nonsensical.

Models under test: {model_1}, {model_2}

Recipe: {r1_name}
  {model_1}: {r1_m1}
  {model_2}: {r1_m2}

Recipe: {r2_name}
  {model_1}: {r2_m1}
  {model_2}: {r2_m2}

Recipe: {r3_name}
  {model_1}: {r3_m1}
  {model_2}: {r3_m2}

Format each line as:
  PASS  recipe_name  ({model})
  FAIL  recipe_name  ({model})

End with a summary line: "Results: X/Y passed, Z failed"
$$;

-- Recipe 11: Debate Arena
-- Two LLM personas argue opposing sides of a topic, then a judge picks the winner.
-- Demonstrates multi-perspective reasoning, adversarial generation, and semantic evaluation.
--
-- Usage:
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama -m gemma3 topic="AI should be open-sourced"
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama -m llama3.2 topic="Remote work is better than office work"
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama topic="Tabs are better than spaces"

CREATE FUNCTION pro_argument(topic TEXT, previous TEXT) RETURNS TEXT
AS $$
You are a skilled debate champion arguing STRONGLY IN FAVOR of the following motion:

Motion: "{topic}"

Your opponent's last argument (or "opening statement" if this is your first turn):
{previous}

Write a focused, persuasive argument supporting the motion. If this is a rebuttal round,
directly address and counter your opponent's points. Be concise (3-5 paragraphs).
Do NOT offer balanced views — you are arguing one side.
$$;

CREATE FUNCTION con_argument(topic TEXT, previous TEXT) RETURNS TEXT
AS $$
You are a skilled debate champion arguing STRONGLY AGAINST the following motion:

Motion: "{topic}"

Your opponent's last argument (or "opening statement" if this is your first turn):
{previous}

Write a focused, persuasive argument opposing the motion. If this is a rebuttal round,
directly address and counter your opponent's points. Be concise (3-5 paragraphs).
Do NOT offer balanced views — you are arguing one side.
$$;

CREATE FUNCTION judge_debate(topic TEXT, pro_history TEXT, con_history TEXT) RETURNS TEXT
AS $$
You are an impartial debate judge evaluating the following debate.

Motion: "{topic}"

--- PRO SIDE (arguing IN FAVOR) ---
{pro_history}

--- CON SIDE (arguing AGAINST) ---
{con_history}

Evaluate the debate on these criteria:
1. Strength of arguments
2. Quality of rebuttals
3. Clarity and persuasiveness

Declare a winner (PRO or CON) and explain your reasoning in 2-3 paragraphs.
$$;

WORKFLOW debate_arena
    INPUT:
        @topic TEXT,
        @max_rounds INTEGER DEFAULT 3,
        @log_dir TEXT DEFAULT 'cookbook/11_debate_arena/logs'
    OUTPUT: @verdict TEXT
DO
    @round := 0
    @pro_history := ''
    @con_history := ''

    LOGGING f'Debate started | topic: {@topic} | rounds: {@max_rounds}' LEVEL INFO

    -- Opening statements
    GENERATE pro_argument(@topic, 'opening statement') INTO @pro
    GENERATE con_argument(@topic, 'opening statement') INTO @con

    @pro_history := @pro
    @con_history := @con

    LOGGING 'Opening statements complete' LEVEL INFO
    CALL write_file(f'{@log_dir}/opening_pro.md', @pro) INTO @_
    CALL write_file(f'{@log_dir}/opening_con.md', @con) INTO @_

    -- Rebuttal rounds
    WHILE @round < @max_rounds DO
        LOGGING f'Round {@round} | pro rebuttal ...' LEVEL DEBUG
        -- Pro rebuts Con's latest point
        GENERATE pro_argument(@topic, @con_history) INTO @pro_rebuttal
        @pro_history := @pro_history || '\n---\n' || @pro_rebuttal
        CALL write_file(f'{@log_dir}/round_{@round}_pro.md', @pro_rebuttal) INTO @_

        LOGGING f'Round {@round} | con rebuttal ...' LEVEL DEBUG
        -- Con rebuts Pro's latest point
        GENERATE con_argument(@topic, @pro_history) INTO @con_rebuttal
        @con_history := @con_history || '\n---\n' || @con_rebuttal
        CALL write_file(f'{@log_dir}/round_{@round}_con.md', @con_rebuttal) INTO @_

        @round := @round + 1
        LOGGING f'Round {@round} complete' LEVEL INFO
    END

    LOGGING 'All rounds done — judge deliberating ...' LEVEL INFO

    -- Judge evaluates both sides
    GENERATE judge_debate(@topic, @pro_history, @con_history) INTO @verdict

    LOGGING f'Verdict ready | rounds={@round}' LEVEL INFO
    CALL write_file(f'{@log_dir}/verdict.md', @verdict) INTO @_
    COMMIT @verdict WITH status = 'complete', rounds = @round

EXCEPTION
    WHEN MaxIterationsReached THEN
        GENERATE judge_debate(@topic, @pro_history, @con_history) INTO @verdict
        COMMIT @verdict WITH status = 'partial'
    WHEN BudgetExceeded THEN
        COMMIT @pro_history WITH status = 'budget_limit'
END

-- Recipe 13: Map-Reduce Summarizer
-- Splits a large document into chunks, summarizes each chunk (map),
-- then combines all summaries into a final summary (reduce).
-- Demonstrates the map-reduce pattern for handling content beyond context limits.
--
-- Usage:
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama -m gemma3 document="$(cat large_doc.txt)"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="$(cat README.md)" style="bullet points"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="long text here..." style="executive brief"
--
-- Tools: chunk_plan(), extract_chunk() are in tools.py (CALL, zero tokens).
--        write_file(), read_file() are SPL built-ins — no --tools flag needed.
--        summarize_chunk(), reduce_summaries(), quality_score(), improve_summary() call the LLM.
--
-- Logs written to @log_dir (default: cookbook/13_map_reduce/logs):
--   chunk_00.md, chunk_01.md, ...    — raw extracted chunks
--   summary_00.md, summary_01.md, ... — per-chunk LLM summaries
--   final_summary.md                  — combined final summary

WORKFLOW map_reduce_summarizer
    INPUT:  
        @document TEXT, 
        @style TEXT,
        @log_dir TEXT DEFAULT 'cookbook/13_map_reduce/logs'
    OUTPUT: @final_summary TEXT
DO
    @chunk_index := 0
    @summaries := []          -- LIST: accumulates per-chunk summaries

    LOGGING 'Starting map-reduce | document length: ' + @document

    -- Step 1: Determine how to split the document (deterministic, 0 tokens)
    CALL chunk_plan(@document) INTO @chunk_count

    LOGGING 'Document split into ' + @chunk_count + ' chunks'

    -- Step 2: MAP — summarize each chunk independently
    WHILE @chunk_index < @chunk_count DO
        CALL extract_chunk(@document, @chunk_index, @chunk_count) INTO @chunk   -- deterministic, 0 tokens

        -- Save raw chunk to logs/
        CALL write_file(f'{@log_dir}/chunk_{@chunk_index}.md', @chunk) INTO @_

        GENERATE summarize_chunk(@chunk, @chunk_index) INTO @chunk_summary

        -- Save chunk summary to logs/
        CALL write_file(f'{@log_dir}/summary_{@chunk_index}.md', @chunk_summary) INTO @_

        LOGGING f'[Chunk {@chunk_index}/{@chunk_count}] summary saved'

        @summaries := list_append(@summaries, @chunk_summary)
        @chunk_index := @chunk_index + 1
    END

    -- Step 3: REDUCE — concat all chunk summaries, then combine into final
    @summaries_text := list_concat(@summaries, '\n')
    GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary

    -- Step 4: Quality check the final summary
    GENERATE quality_score(@final_summary, @document) INTO @score

    EVALUATE @score
        WHEN > 0.7 THEN
            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Final summary saved to {@log_dir}/final_summary.md (score={@score})'
            COMMIT @final_summary WITH status = 'complete', chunks = @chunk_count

            GENERATE improve_summary(@final_summary, @summaries_text) INTO @final_summary
            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Improved summary saved to {@log_dir}/final_summary.md (score={@score})'
            COMMIT @final_summary WITH status = 'refined', chunks = @chunk_count
    END

EXCEPTION

        GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary
        CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
        COMMIT @final_summary WITH status = 'partial'

-- The agent solves a problem, then reflects on its own reasoning to catch errors.
-- If reflection finds issues, it corrects and re-reflects until confident.
-- Demonstrates meta-cognitive loops and self-assessment patterns.
--
-- Usage:
--   spl run cookbook/16_reflection/reflection.spl --adapter ollama -m gemma3 problem="What are the trade-offs of microservices vs monoliths?"

        @problem TEXT,
        @max_reflections Number DEFAULT 3,
        @log_dir TEXT DEFAULT 'cookbook/16_reflection/logs'

    @confidence := 0

    LOGGING f'Reflection agent started | max_reflections={@max_reflections}' LEVEL INFO

    -- Initial attempt at solving the problem

    LOGGING 'Initial solution ready' LEVEL INFO

    -- Reflection loop: assess and improve

        LOGGING f'Reflection iteration {@iteration} ...' LEVEL DEBUG
        -- Reflect: identify errors, gaps, or weak reasoning
        GENERATE reflect(@problem, @answer) INTO @reflection
        CALL write_file(f'{@log_dir}/reflection_{@iteration}.md', @reflection) INTO @_

        -- Score confidence in the answer

                -- Extract specific issues to fix
                GENERATE extract_issues(@reflection) INTO @issues

                -- Correct the answer based on identified issues
                GENERATE correct(@answer, @issues, @problem) INTO @answer
                @iteration := @iteration + 1
                CALL write_file(f'{@log_dir}/answer_{@iteration}.md', @answer) INTO @_

    END

    -- If loop exhausted, commit best effort with final confidence
    LOGGING f'Max reflections reached | confidence={@confidence}' LEVEL WARN
    CALL write_file(f'{@log_dir}/final.md', @answer) INTO @_
    COMMIT @answer WITH status = 'best_effort', confidence = @confidence, reflections = @iteration

-- Recipe 21: Multi-Model Pipeline
-- Showcases GENERATE ... USING MODEL per-step model selection
-- and semantic WHILE loop with variable context.
--
-- Each step can target a different model — the right model for each task.
-- Usage:
--   spl run cookbook/21_multi_model_pipeline/multi_model.spl --adapter ollama topic="climate change"

CREATE FUNCTION research(topic TEXT) RETURNS TEXT 
AS $$

You are a research specialist.
Provide key facts, statistics, and recent developments about: {topic}
Focus on factual accuracy. Be thorough but concise.
$$;

CREATE FUNCTION analyze(facts TEXT) RETURNS TEXT 
AS $$

You are a data analyst.
Analyze the following research and identify the 3 most significant insights:

{facts}

For each insight, explain why it matters and rate its significance (1-10).
$$;

CREATE FUNCTION write_summary(analysis TEXT) RETURNS TEXT 
AS $$

You are a professional writer.
Write a clear, engaging 2-paragraph summary based on this analysis:

{analysis}

The first paragraph should present the key findings.
The second paragraph should discuss implications and outlook.
$$;

CREATE FUNCTION quality_check(text TEXT) RETURNS TEXT 
AS $$

You are a quality reviewer.
Rate the following text on a scale of 0.0 to 1.0 for:
- Clarity
- Accuracy
- Completeness

Return ONLY a single number (the average score).

Text: {text}
$$;

WORKFLOW multi_model_pipeline
  INPUT:
    @topic TEXT,
    @max_iterations INT DEFAULT 3,
    @log_dir TEXT DEFAULT 'cookbook/21_multi_model_pipeline/logs'
  OUTPUT: @final TEXT
DO
  LOGGING f'Multi-model pipeline | topic={@topic}' LEVEL INFO

  -- Step 1: Research with a fast model (good for factual retrieval)
  GENERATE research(@topic) USING MODEL 'gemma3' INTO @facts
  LOGGING 'Research complete' LEVEL DEBUG
  CALL write_file(f'{@log_dir}/research.md', @facts) INTO @_

  -- Step 2: Analysis with a reasoning model
  GENERATE analyze(@facts) USING MODEL 'gemma3' INTO @analysis
  LOGGING 'Analysis complete' LEVEL DEBUG
  CALL write_file(f'{@log_dir}/analysis.md', @analysis) INTO @_

  -- Step 3: Writing with a creative model
  GENERATE write_summary(@analysis) USING MODEL 'gemma3' INTO @draft
  LOGGING 'Initial draft ready' LEVEL INFO
  CALL write_file(f'{@log_dir}/draft_0.md', @draft) INTO @_

  -- Step 4: Quality loop — refine until score > 0.7 (max 3 iterations)
  @iteration := 0
  @quality := 0

  WHILE @iteration < @max_iterations DO
    LOGGING f'Quality check | iteration={@iteration}' LEVEL DEBUG
    GENERATE quality_check(@draft) INTO @quality
    LOGGING f'Quality score: {@quality}' LEVEL DEBUG

    EVALUATE @quality
      WHEN > 0.7 THEN
        LOGGING f'Quality threshold met | score={@quality}' LEVEL INFO
        CALL write_file(f'{@log_dir}/final.md', @draft) INTO @_
        COMMIT @draft WITH status = 'high_quality', score = @quality
      ELSE
        GENERATE write_summary(@analysis) INTO @draft
        @iteration := @iteration + 1
        CALL write_file(f'{@log_dir}/draft_{@iteration}.md', @draft) INTO @_
    END
  END

  -- Fallback: commit whatever we have
  LOGGING f'Max iterations reached | final score={@quality}' LEVEL WARN
  CALL write_file(f'{@log_dir}/final.md', @draft) INTO @_
  COMMIT @draft WITH status = 'max_iterations', score = @quality

EXCEPTION
  WHEN MaxIterationsReached THEN
    COMMIT @draft WITH status = 'partial'
  WHEN ModelOverloaded THEN
    COMMIT @draft WITH status = 'model_overloaded'
END
