-- Recipe 04: Model Showdown
-- Same prompt to multiple Ollama models — compare responses side-by-side.
-- Uses parallel CTEs instead of a bash loop for a pure-SPL implementation.
--
-- Usage:
--   spl2 run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="What is the meaning of life?"
--   spl2 run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="Explain recursion in 3 sentences" \
--       model_1=gemma3 model_2=llama3.2 model_3=mistral

CREATE FUNCTION compare_responses(
    prompt TEXT,
    model_1 TEXT, answer_1 TEXT,
    model_2 TEXT, answer_2 TEXT,
    model_3 TEXT, answer_3 TEXT
) RETURNS TEXT AS $$
You are a neutral evaluator comparing responses from three AI models.

Prompt given to all models:
"{prompt}"

=== {model_1} ===
{answer_1}

=== {model_2} ===
{answer_2}

=== {model_3} ===
{answer_3}

For each model, briefly assess:
- Response quality and accuracy
- Key strengths or weaknesses
- Notable differences from the others

Conclude with which model gave the most helpful answer and why.
$$;

WORKFLOW model_showdown
    INPUT:
        @prompt  TEXT DEFAULT 'What is the meaning of life?',
        @model_1 TEXT DEFAULT 'gemma3',
        @model_2 TEXT DEFAULT 'llama3.2',
        @model_3 TEXT DEFAULT 'mistral'
    OUTPUT: @comparison TEXT
DO
    -- Run the same prompt against all three models in parallel CTEs
    WITH
        response_1 AS (
            PROMPT ask_model_1
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_1
        ),
        response_2 AS (
            PROMPT ask_model_2
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_2
        ),
        response_3 AS (
            PROMPT ask_model_3
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_3
        )
    SELECT
        response_1.answer AS answer_1,
        response_2.answer AS answer_2,
        response_3.answer AS answer_3
    INTO @answer_1, @answer_2, @answer_3

    -- Judge all three responses together
    GENERATE compare_responses(
        @prompt,
        @model_1, @answer_1,
        @model_2, @answer_2,
        @model_3, @answer_3
    ) INTO @comparison

    COMMIT @comparison WITH
        status   = 'complete',
        model_1  = @model_1,
        model_2  = @model_2,
        model_3  = @model_3

EXCEPTION
    WHEN GenerationError THEN
        COMMIT 'One or more models failed to respond.' WITH status = 'error'
END

-- SPL 2.0: Self-Refine Pattern
-- Iteratively improves output through critique and refinement

WORKFLOW self_refine
  INPUT: @task text
  OUTPUT: @result text
DO
  @iteration := 0
  @max_iterations := 5

  -- Initial draft
  GENERATE draft(@task) INTO @current

  -- Iterative refinement loop
  WHILE @iteration < @max_iterations DO
    GENERATE critique(@current) INTO @feedback

    EVALUATE @feedback
      WHEN 'satisfactory' THEN
        COMMIT @current WITH status = 'complete', iterations = @iteration
      OTHERWISE
        GENERATE refined(@current, @feedback) INTO @current
        @iteration := @iteration + 1
    END
  END

  -- If loop exhausted, commit best effort
  COMMIT @current WITH status = 'max_iterations', iterations = @iteration

EXCEPTION
  WHEN MaxIterationsReached THEN
    COMMIT @current WITH status = 'partial'
  WHEN BudgetExceeded THEN
    COMMIT @current WITH status = 'budget_limit'
END

-- SPL 2.0: ReAct Agent Pattern
-- Reasoning + Acting loop with tool use

WORKFLOW react_agent
  INPUT: @task text
  OUTPUT: @answer text
DO
  @iteration := 0
  @history := ''
  @context := ''

  WHILE @iteration < 10 DO

    -- Reasoning step
    GENERATE thought(@task, @history, @context) INTO @current_thought

    -- Action decision
    GENERATE action_decision(@current_thought) INTO @action

    -- Execute action based on classification
    EVALUATE @action
      WHEN 'search' THEN
        CALL search_tool(@action) INTO @observation
      WHEN 'calculate' THEN
        CALL calculator(@action) INTO @observation
      WHEN 'answer' THEN
        GENERATE final_answer(@history) INTO @answer
        COMMIT @answer WITH status = 'complete'
    END

    -- Accumulate history
    @history := @history + @current_thought
    @iteration := @iteration + 1

  END

EXCEPTION
  WHEN MaxIterationsReached THEN
    GENERATE best_answer(@history) INTO @answer
    COMMIT @answer WITH status = 'partial'
  WHEN HallucinationDetected THEN
    RETRY WITH temperature = 0.1
END

WORKFLOW safe_generation

  OUTPUT: @result text
  SECURITY: CLASSIFICATION: internal
DO
  GENERATE response(@prompt) INTO @result

  GENERATE quality_assess(@result) INTO @quality

    WHEN 'high_quality' THEN
      COMMIT @result WITH status = 'high_quality'
    WHEN 'acceptable' THEN

-- A planner agent decomposes a complex task into steps, then an executor runs each step.
-- Demonstrates agentic planning, sequential execution, and progress tracking.
--
-- Usage:
--   spl2 run cookbook/12_plan_and_execute/plan_execute.spl --adapter ollama -m gemma3 task="Build a REST API for a todo app"
--   spl2 run cookbook/12_plan_and_execute/plan_execute.spl --adapter ollama task="Set up a CI/CD pipeline for a Python project"

WORKFLOW plan_and_execute
    INPUT: @task TEXT
    OUTPUT: @final_report TEXT
DO
    @step_index := 0
    @results := ''

    -- Phase 1: Planning — decompose task into numbered steps
    GENERATE plan(@task) INTO @plan

    -- Phase 2: Count steps in plan
    GENERATE count_steps(@plan) INTO @step_count

    -- Phase 3: Execute each step sequentially
    WHILE @step_index < @step_count DO
        -- Extract current step from plan
        GENERATE extract_step(@plan, @step_index) INTO @current_step

        -- Execute the step with context from prior results
        GENERATE execute_step(@current_step, @results) INTO @step_result

        -- Validate step output
        GENERATE validate_step(@current_step, @step_result) INTO @validation

        EVALUATE @validation
            WHEN 'failed' THEN
                -- Re-plan remaining steps given the failure
                GENERATE replan(@task, @plan, @step_index, @step_result) INTO @plan
                GENERATE count_steps(@plan) INTO @step_count
                @step_index := 0
                @results := ''
            OTHERWISE
                @results := @results + '\n## Step ' + @step_index + '\n' + @step_result
                @step_index := @step_index + 1
        END
    END

    -- Phase 4: Synthesize final report
    GENERATE synthesize(@task, @results) INTO @final_report

    COMMIT @final_report WITH status = 'complete', steps_executed = @step_index

EXCEPTION
    WHEN MaxIterationsReached THEN
        GENERATE synthesize(@task, @results) INTO @final_report

    WHEN BudgetExceeded THEN

-- Recipe 13: Map-Reduce Summarizer
-- Splits a large document into chunks, summarizes each chunk (map),
-- then combines all summaries into a final summary (reduce).
-- Demonstrates the map-reduce pattern for handling content beyond context limits.
--
-- Usage:
--   spl2 run cookbook/13_map_reduce/map_reduce.spl --adapter ollama -m gemma3 document="$(cat large_doc.txt)"
--   spl2 run cookbook/13_map_reduce/map_reduce.spl --adapter ollama document="$(cat README.md)" style="bullet points"
--   spl2 run cookbook/13_map_reduce/map_reduce.spl --adapter ollama document="long text here..." style="executive brief"

WORKFLOW map_reduce_summarizer
    INPUT: @document TEXT, @style TEXT
    OUTPUT: @final_summary TEXT
DO
    @chunk_index := 0
    @summaries := ''

    -- Step 1: Determine how to split the document
    GENERATE chunk_plan(@document) INTO @chunk_count

    -- Step 2: MAP — summarize each chunk independently
    WHILE @chunk_index < @chunk_count DO
        GENERATE extract_chunk(@document, @chunk_index, @chunk_count) INTO @chunk

        GENERATE summarize_chunk(@chunk, @chunk_index) INTO @chunk_summary

        @summaries := @summaries + '\n[Chunk ' + @chunk_index + ']: ' + @chunk_summary
        @chunk_index := @chunk_index + 1
    END

    -- Step 3: REDUCE — combine all chunk summaries into one
    GENERATE reduce_summaries(@summaries, @style) INTO @final_summary

    -- Step 4: Quality check the final summary
    GENERATE quality_score(@final_summary, @document) INTO @score

RETURNS TEXT

            COMMIT @calibrated
        OTHERWISE

    END

def assemble_sentiment_report(data):
    """
    Assembles and processes the given product reviews data.
    
    Args:
        data (list): A list of strings containing product reviews.
        
    Returns:
        dict: A dictionary where keys are unique words in each review, 
              and values represent their sentiment scores or topic scores.
    """

    # Initialize an empty dictionary to store results
    result = defaultdict(dict)

    for i, review in enumerate(data):
        # Clean the review by replacing non-alphanumeric characters with a space
        cleaned_review = re.sub(r'[^a-zA-Z0-9\s]', '', review)
        
        # Split the review into words
        word_list = cleaned_review.split()
        
        # Count the frequency of each word
        word_freq = Counter(word_list)
        
        # Filter out words with a frequency of more than 1
        filtered_word_freq = {word: freq for word, freq in word_freq.items() if freq == 1}
        
        result[i].update(filtered_word_freq)

    return result

  WHEN 'sentence'  THEN 'One sentence only. Maximum 25 words. Capture the single most important idea.'
  WHEN 'paragraph' THEN '3-5 sentences. Cover main points and key supporting evidence. No examples or details.'
  WHEN 'page'      THEN '2-3 paragraphs. Include context, main argument, evidence, and implications.'

END
$$;
