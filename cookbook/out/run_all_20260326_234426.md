-- Recipe 04: Model Showdown
-- Same prompt to multiple Ollama models — compare responses side-by-side.
-- Uses parallel CTEs instead of a bash loop for a pure-SPL implementation.
--
-- Usage:
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="What is the meaning of life?"
--   spl run cookbook/04_model_showdown/showdown.spl --adapter ollama \
--       prompt="Explain recursion in 3 sentences" \
--       model_1=gemma3 model_2=llama3.2 model_3=mistral

CREATE FUNCTION compare_responses(
    prompt TEXT,
    model_1 TEXT, answer_1 TEXT,
    model_2 TEXT, answer_2 TEXT,
    model_3 TEXT, answer_3 TEXT
) RETURNS TEXT
AS $$
You are a neutral evaluator comparing responses from three AI models.

Prompt given to all models: "{prompt}"

First, reproduce each model's raw response exactly as given, formatted like this:

=== {model_1} ===
{answer_1}

=== {model_2} ===
{answer_2}

=== {model_3} ===
{answer_3}

Then provide your evaluation. For each model briefly assess:
- Response quality
- Key strengths or weaknesses

Conclude with which model gave the most helpful answer and why.
$$;

WORKFLOW model_showdown
    INPUT:
        @prompt  TEXT DEFAULT 'What is the meaning of life?',
        @model_1 TEXT DEFAULT 'gemma3',
        @model_2 TEXT DEFAULT 'phi4',
        @model_3 TEXT DEFAULT 'qwen2.5'
    OUTPUT: @comparison TEXT
DO
    -- Run the same prompt against all three models in parallel CTEs
    WITH
        response_1 AS (
            PROMPT ask_model_1
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_1
        ),
        response_2 AS (
            PROMPT ask_model_2
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_2
        ),
        response_3 AS (
            PROMPT ask_model_3
            SELECT
                system_role('You are a helpful, knowledgeable assistant.'),
                @prompt AS prompt
            GENERATE answer(prompt)
            USING MODEL @model_3
        )
    SELECT
        response_1.answer AS answer_1,
        response_2.answer AS answer_2,
        response_3.answer AS answer_3
    INTO @answer_1, @answer_2, @answer_3

    -- Judge all three responses together
    GENERATE compare_responses(
        @prompt,
        @model_1, @answer_1,
        @model_2, @answer_2,
        @model_3, @answer_3
    ) INTO @comparison

    COMMIT @comparison WITH
        status   = 'complete',
        model_1  = @model_1,
        model_2  = @model_2,
        model_3  = @model_3

EXCEPTION
    WHEN GenerationError THEN
        COMMIT 'One or more models failed to respond.' WITH status = 'error'
END

-- SPL 2.0: Self-Refine Pattern
-- Iteratively improves output through critique and refinement

WORKFLOW self_refine
  INPUT:
    @task TEXT,
    @max_iterations INTEGER DEFAULT 5,
    @log_dir TEXT DEFAULT 'cookbook/05_self_refine/logs'
  OUTPUT: @result TEXT
DO
  @iteration := 0

  LOGGING f'Self-refine started | max_iterations={@max_iterations}' LEVEL INFO

  -- Initial draft
  GENERATE draft(@task) INTO @current
  LOGGING 'Initial draft ready' LEVEL INFO
  CALL write_file(f'{@log_dir}/draft_0.md', @current) INTO @_

  -- Iterative refinement loop
  WHILE @iteration < @max_iterations DO
    LOGGING f'Iteration {@iteration} | critiquing ...' LEVEL DEBUG
    GENERATE critique(@current) INTO @feedback
    CALL write_file(f'{@log_dir}/feedback_{@iteration}.md', @feedback) INTO @_

    EVALUATE @feedback
      WHEN 'satisfactory' THEN
        LOGGING f'Satisfactory at iteration {@iteration}' LEVEL INFO
        CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
        COMMIT @current WITH status = 'complete', iterations = @iteration
      ELSE
        GENERATE refined(@current, @feedback) INTO @current
        @iteration := @iteration + 1
        CALL write_file(f'{@log_dir}/draft_{@iteration}.md', @current) INTO @_
        LOGGING f'Refined | iteration={@iteration}' LEVEL DEBUG
    END
  END

  -- If loop exhausted, commit best effort
  LOGGING f'Max iterations reached | iterations={@iteration}' LEVEL WARN
  CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
  COMMIT @current WITH status = 'max_iterations', iterations = @iteration

EXCEPTION
  WHEN MaxIterationsReached THEN
    CALL write_file(f'{@log_dir}/final.md', @current) INTO @_
    COMMIT @current WITH status = 'partial'
  WHEN BudgetExceeded THEN
    COMMIT @current WITH status = 'budget_limit'
END

-- SPL 2.0: ReAct Agent — Population Growth
-- Fetch a country's population for the past two years via WebSearch,
-- then compute year-over-year growth rate using a Python tool (no LLM math).
--
-- Usage:
--   spl run cookbook/06_react_agent/react_agent.spl \
--       --adapter claude_cli -m claude-sonnet-4-6 \
--       --allowed-tools WebSearch \
--       --tools cookbook/06_react_agent/tools.py \
--       country="France"

-- Tight prompt for search_population: return only the number, no narration.
CREATE FUNCTION search_population(country TEXT, year INT) RETURNS TEXT 
AS $$
Use WebSearch to find the total population of {country} for the year {year}.

Return ONLY the population as a plain integer with no commas, spaces, or text.
Example of valid output: 1409670000
No explanation. No units. No markdown. Just the number.
$$;

-- SPL PROCEDURE: search the web for a country's population in a specific year.
PROCEDURE search_population(country TEXT, year INT)
RETURNS TEXT
DO
  GENERATE search_population(country, year) INTO @result
  COMMIT @result
END

-- calc_growth_rate is a Python tool registered from tools.py via --tools.
-- CALL routes to Python directly — deterministic, zero LLM cost.

-- Tight prompt for growth_report: plain narrative, no tool use, no file writes.
CREATE FUNCTION growth_report(country TEXT, year_prev TEXT, pop_prev TEXT, year_curr TEXT, pop_curr TEXT, growth_rate TEXT) 
RETURNS TEXT 
AS $$
Write a concise 2-3 sentence population growth report for {country}.

Data:
- {year_prev} population: {pop_prev}
- {year_curr} population: {pop_curr}
- Year-over-year growth rate: {growth_rate}%

Plain prose only. No markdown headers. No file creation. No tool use. Just the narrative.
$$;

WORKFLOW population_growth
  INPUT:
    @country TEXT DEFAULT 'China',
    @year_curr INTEGER DEFAULT 2023
  OUTPUT: @report TEXT
DO
  @year_prev := @year_curr - 1

  LOGGING f'Population growth | country={@country} years={@year_prev}-{@year_curr}' LEVEL INFO

  -- Step 1: Fetch population for each year via WebSearch
  CALL search_population(@country, @year_prev) INTO @pop_prev
  CALL search_population(@country, @year_curr) INTO @pop_curr
  LOGGING f'Population fetched | {@year_prev}={@pop_prev} {@year_curr}={@pop_curr}' LEVEL DEBUG

  -- Step 2: Compute growth rate — routed to Python tool, not LLM
  CALL calc_growth_rate(@pop_prev, @pop_curr) INTO @growth_rate
  LOGGING f'Growth rate computed: {@growth_rate}%' LEVEL INFO

  -- Step 3: Compose the final human-readable report
  GENERATE growth_report(@country, @year_prev, @pop_prev, @year_curr, @pop_curr, @growth_rate) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN SearchFailed THEN
    GENERATE search_error_report(@country) INTO @report
    COMMIT @report WITH status = 'error'
END

You are a test reporter. Review the outputs below and produce a concise pass/fail report.

A result is PASS if the output is a non-empty, coherent response.
A result is FAIL if the output is empty, contains an error, or is clearly nonsensical.

Models under test: {model_1}, {model_2}

Recipe: {r1_name}

Recipe: {r2_name}

Format each line as:

-- Recipe 11: Debate Arena
-- Two LLM personas argue opposing sides of a topic, then a judge picks the winner.
-- Demonstrates multi-perspective reasoning, adversarial generation, and semantic evaluation.
--
-- Usage:
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama -m gemma3 topic="AI should be open-sourced"
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama -m llama3.2 topic="Remote work is better than office work"
--   spl run cookbook/11_debate_arena/debate.spl --adapter ollama topic="Tabs are better than spaces"

CREATE FUNCTION pro_argument(topic TEXT, previous TEXT) RETURNS TEXT
AS $$
You are a skilled debate champion arguing STRONGLY IN FAVOR of the following motion:

Motion: "{topic}"

Your opponent's last argument (or "opening statement" if this is your first turn):
{previous}

Write a focused, persuasive argument supporting the motion. If this is a rebuttal round,
directly address and counter your opponent's points. Be concise (3-5 paragraphs).
Do NOT offer balanced views — you are arguing one side.
$$;

CREATE FUNCTION con_argument(topic TEXT, previous TEXT) RETURNS TEXT
AS $$
You are a skilled debate champion arguing STRONGLY AGAINST the following motion:

Motion: "{topic}"

Your opponent's last argument (or "opening statement" if this is your first turn):
{previous}

Write a focused, persuasive argument opposing the motion. If this is a rebuttal round,
directly address and counter your opponent's points. Be concise (3-5 paragraphs).
Do NOT offer balanced views — you are arguing one side.
$$;

CREATE FUNCTION judge_debate(topic TEXT, pro_history TEXT, con_history TEXT) RETURNS TEXT
AS $$
You are an impartial debate judge evaluating the following debate.

Motion: "{topic}"

--- PRO SIDE (arguing IN FAVOR) ---
{pro_history}

--- CON SIDE (arguing AGAINST) ---
{con_history}

Evaluate the debate on these criteria:
1. Strength of arguments
2. Quality of rebuttals
3. Clarity and persuasiveness

Declare a winner (PRO or CON) and explain your reasoning in 2-3 paragraphs.
$$;

WORKFLOW debate_arena
    INPUT:
        @topic TEXT,
        @max_rounds INTEGER DEFAULT 3,
        @log_dir TEXT DEFAULT 'cookbook/11_debate_arena/logs'
    OUTPUT: @verdict TEXT
DO
    @round := 0
    @pro_history := ''
    @con_history := ''

    LOGGING f'Debate started | topic: {@topic} | rounds: {@max_rounds}' LEVEL INFO

    -- Opening statements
    GENERATE pro_argument(@topic, 'opening statement') INTO @pro
    GENERATE con_argument(@topic, 'opening statement') INTO @con

    @pro_history := @pro
    @con_history := @con

    LOGGING 'Opening statements complete' LEVEL INFO
    CALL write_file(f'{@log_dir}/opening_pro.md', @pro) INTO @_
    CALL write_file(f'{@log_dir}/opening_con.md', @con) INTO @_

    -- Rebuttal rounds
    WHILE @round < @max_rounds DO
        LOGGING f'Round {@round} | pro rebuttal ...' LEVEL DEBUG
        -- Pro rebuts Con's latest point
        GENERATE pro_argument(@topic, @con_history) INTO @pro_rebuttal
        @pro_history := @pro_history || '\n---\n' || @pro_rebuttal
        CALL write_file(f'{@log_dir}/round_{@round}_pro.md', @pro_rebuttal) INTO @_

        LOGGING f'Round {@round} | con rebuttal ...' LEVEL DEBUG
        -- Con rebuts Pro's latest point
        GENERATE con_argument(@topic, @pro_history) INTO @con_rebuttal
        @con_history := @con_history || '\n---\n' || @con_rebuttal
        CALL write_file(f'{@log_dir}/round_{@round}_con.md', @con_rebuttal) INTO @_

        @round := @round + 1
        LOGGING f'Round {@round} complete' LEVEL INFO
    END

    LOGGING 'All rounds done — judge deliberating ...' LEVEL INFO

    -- Judge evaluates both sides
    GENERATE judge_debate(@topic, @pro_history, @con_history) INTO @verdict

    LOGGING f'Verdict ready | rounds={@round}' LEVEL INFO
    CALL write_file(f'{@log_dir}/verdict.md', @verdict) INTO @_
    COMMIT @verdict WITH status = 'complete', rounds = @round

EXCEPTION
    WHEN MaxIterationsReached THEN
        GENERATE judge_debate(@topic, @pro_history, @con_history) INTO @verdict
        COMMIT @verdict WITH status = 'partial'
    WHEN BudgetExceeded THEN
        COMMIT @pro_history WITH status = 'budget_limit'
END

-- Recipe 13: Map-Reduce Summarizer
-- Splits a large document into chunks, summarizes each chunk (map),
-- then combines all summaries into a final summary (reduce).
-- Demonstrates the map-reduce pattern for handling content beyond context limits.
--
-- Usage:
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama -m gemma3 document="$(cat large_doc.txt)"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="$(cat README.md)" style="bullet points"
--   spl run cookbook/13_map_reduce/map_reduce.spl --tools cookbook/13_map_reduce/tools.py --adapter ollama document="long text here..." style="executive brief"
--
-- Tools: chunk_plan(), extract_chunk() are in tools.py (CALL, zero tokens).
--        write_file(), read_file() are SPL built-ins — no --tools flag needed.
--        summarize_chunk(), reduce_summaries(), quality_score(), improve_summary() call the LLM.
--
-- Logs written to @log_dir (default: cookbook/13_map_reduce/logs):
--   chunk_00.md, chunk_01.md, ...    — raw extracted chunks
--   summary_00.md, summary_01.md, ... — per-chunk LLM summaries
--   final_summary.md                  — combined final summary

WORKFLOW map_reduce_summarizer
    INPUT:  
        @document TEXT, 
        @style TEXT,
        @log_dir TEXT DEFAULT 'cookbook/13_map_reduce/logs'
    OUTPUT: @final_summary TEXT
DO
    @chunk_index := 0
    @summaries := []          -- LIST: accumulates per-chunk summaries

    LOGGING 'Starting map-reduce | document length: ' + @document

    -- Step 1: Determine how to split the document (deterministic, 0 tokens)
    CALL chunk_plan(@document) INTO @chunk_count

    LOGGING 'Document split into ' + @chunk_count + ' chunks'

    -- Step 2: MAP — summarize each chunk independently
    WHILE @chunk_index < @chunk_count DO
        CALL extract_chunk(@document, @chunk_index, @chunk_count) INTO @chunk   -- deterministic, 0 tokens

        -- Save raw chunk to logs/
        CALL write_file(f'{@log_dir}/chunk_{@chunk_index}.md', @chunk) INTO @_

        GENERATE summarize_chunk(@chunk, @chunk_index) INTO @chunk_summary

        -- Save chunk summary to logs/
        CALL write_file(f'{@log_dir}/summary_{@chunk_index}.md', @chunk_summary) INTO @_

        LOGGING f'[Chunk {@chunk_index}/{@chunk_count}] summary saved'

        @summaries := list_append(@summaries, @chunk_summary)
        @chunk_index := @chunk_index + 1
    END

    -- Step 3: REDUCE — concat all chunk summaries, then combine into final
    @summaries_text := list_concat(@summaries, '\n')
    GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary

    -- Step 4: Quality check the final summary
    GENERATE quality_score(@final_summary, @document) INTO @score

    EVALUATE @score
        WHEN > 0.7 THEN
            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Final summary saved to {@log_dir}/final_summary.md (score={@score})'

            CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_
            LOGGING f'Improved summary saved to {@log_dir}/final_summary.md (score={@score})'
            COMMIT @final_summary WITH status = 'refined', chunks = @chunk_count
    END

        GENERATE reduce_summaries(@summaries_text, @style) INTO @final_summary
        CALL write_file(f'{@log_dir}/final_summary.md', @final_summary) INTO @_

-- The agent solves a problem, then reflects on its own reasoning to catch errors.
-- If reflection finds issues, it corrects and re-reflects until confident.
-- Demonstrates meta-cognitive loops and self-assessment patterns.
--

--   spl run cookbook/16_reflection/reflection.spl --adapter ollama -m gemma3 problem="What are the trade-offs of microservices vs monoliths?"

        @problem TEXT,

    @confidence := 0

                -- Extract specific issues to fix
                GENERATE extract_issues(@reflection) INTO @issues

                -- Correct the answer based on identified issues

                @iteration := @iteration + 1
                CALL write_file(f'{@log_dir}/answer_{@iteration}.md', @answer) INTO @_

    LOGGING f'Max reflections reached | confidence={@confidence}' LEVEL WARN

    COMMIT @answer WITH status = 'best_effort', confidence = @confidence, reflections = @iteration

-- Recipe 21: Multi-Model Pipeline
-- Showcases GENERATE ... USING MODEL per-step model selection
-- and semantic WHILE loop with variable context.
--
-- Each step can target a different model — the right model for each task.
-- Usage:
--   spl run cookbook/21_multi_model_pipeline/multi_model.spl --adapter ollama topic="climate change"

CREATE FUNCTION research(topic TEXT) RETURNS TEXT 
AS $$

You are a research specialist.
Provide key facts, statistics, and recent developments about: {topic}
Focus on factual accuracy. Be thorough but concise.
$$;

CREATE FUNCTION analyze(facts TEXT) RETURNS TEXT 
AS $$

You are a data analyst.
Analyze the following research and identify the 3 most significant insights:

{facts}

For each insight, explain why it matters and rate its significance (1-10).
$$;

CREATE FUNCTION write_summary(analysis TEXT) RETURNS TEXT 
AS $$

You are a professional writer.
Write a clear, engaging 2-paragraph summary based on this analysis:

{analysis}

The first paragraph should present the key findings.
The second paragraph should discuss implications and outlook.
$$;

CREATE FUNCTION quality_check(text TEXT) RETURNS TEXT 
AS $$

You are a quality reviewer.
Rate the following text on a scale of 0.0 to 1.0 for:
- Clarity
- Accuracy
- Completeness

Return ONLY a single number (the average score).

Text: {text}
$$;

WORKFLOW multi_model_pipeline
  INPUT:
    @topic TEXT,
    @max_iterations INT DEFAULT 3,
    @log_dir TEXT DEFAULT 'cookbook/21_multi_model_pipeline/logs'
  OUTPUT: @final TEXT
DO
  LOGGING f'Multi-model pipeline | topic={@topic}' LEVEL INFO

  -- Step 1: Research with a fast model (good for factual retrieval)
  GENERATE research(@topic) USING MODEL 'gemma3' INTO @facts
  LOGGING 'Research complete' LEVEL DEBUG
  CALL write_file(f'{@log_dir}/research.md', @facts) INTO @_

  -- Step 2: Analysis with a reasoning model
  GENERATE analyze(@facts) USING MODEL 'gemma3' INTO @analysis
  LOGGING 'Analysis complete' LEVEL DEBUG
  CALL write_file(f'{@log_dir}/analysis.md', @analysis) INTO @_

  -- Step 3: Writing with a creative model
  GENERATE write_summary(@analysis) USING MODEL 'gemma3' INTO @draft
  LOGGING 'Initial draft ready' LEVEL INFO
  CALL write_file(f'{@log_dir}/draft_0.md', @draft) INTO @_

  -- Step 4: Quality loop — refine until score > 0.7 (max 3 iterations)
  @iteration := 0
  @quality := 0

  WHILE @iteration < @max_iterations DO
    LOGGING f'Quality check | iteration={@iteration}' LEVEL DEBUG
    GENERATE quality_check(@draft) INTO @quality
    LOGGING f'Quality score: {@quality}' LEVEL DEBUG

    EVALUATE @quality
      WHEN > 0.7 THEN
        LOGGING f'Quality threshold met | score={@quality}' LEVEL INFO
        CALL write_file(f'{@log_dir}/final.md', @draft) INTO @_
        COMMIT @draft WITH status = 'high_quality', score = @quality
      ELSE
        GENERATE write_summary(@analysis) INTO @draft
        @iteration := @iteration + 1
        CALL write_file(f'{@log_dir}/draft_{@iteration}.md', @draft) INTO @_
    END
  END

  -- Fallback: commit whatever we have
  LOGGING f'Max iterations reached | final score={@quality}' LEVEL WARN
  CALL write_file(f'{@log_dir}/final.md', @draft) INTO @_
  COMMIT @draft WITH status = 'max_iterations', score = @quality

EXCEPTION
  WHEN MaxIterationsReached THEN
    COMMIT @draft WITH status = 'partial'
  WHEN ModelOverloaded THEN
    COMMIT @draft WITH status = 'model_overloaded'
END

RETURNS TEXT
AS $$
Rate the response on a scale of 0–10 for each criterion:
1. Clarity (0-10): Is it easy to understand?
2. Completeness (0-10): Does it fully address the task?
3. Relevance (0-10): Is every sentence on-topic?
4. Engagement (0-10): Is it interesting to read?

Return valid JSON only — no prose outside the JSON:
{"clarity": N, "completeness": N, "relevance": N, "engagement": N, "total": N, "rationale": "one concise sentence"}
$$;

            @prompt_a         TEXT DEFAULT '',

    -- Step 1: Load experiment from catalog (deterministic — no LLM)
    -- Sets @experiment_context with task, prompt_a, prompt_b, and notes.
    -- If experiment_id is blank, the context will be empty and @task/@prompt_a/@prompt_b are used directly.
    CALL load_experiment(@experiment_id) INTO @experiment_context

        @task,
        @prompt_a,
        @experiment_context
    ) INTO @response_a

    -- Step 3: Generate Variant B response
    GENERATE run_variant_b(
        @task,
        @prompt_b,
        @experiment_context
    ) INTO @response_b

    -- Step 4: Score both variants against the rubric (LLM)
    GENERATE evaluate_response(@response_a, @task, scoring_rubric()) INTO @score_a_json
    GENERATE evaluate_response(@response_b, @task, scoring_rubric()) INTO @score_b_json

    CALL extract_score_total(@score_b_json) INTO @score_b

        WHEN > @winner_threshold THEN
            SET @winner_output = @response_a
            COMMIT @winner_output WITH
                winner    = 'A',
                score_a   = @score_a,
                score_b   = @score_b,

                winner  = 'tie',
                score_a = @score_a,

CREATE FUNCTION support_categories()
RETURNS TEXT
AS $$
Ticket categories and their routing:
- billing:      finance team, SLA 4h,  high priority
                Covers: duplicate charges, wrong amounts, refund requests, invoice issues
- account:      auth team,    SLA 2h,  high priority
                Covers: login failures, password reset, account locked, 2FA issues
- shipping:     fulfillment,  SLA 6h,  high priority
                Covers: delayed delivery, wrong item received, missing package, tracking problems
- technical:    engineering,  SLA 8h,  medium priority
                Covers: app bugs, API errors, integration failures, data export issues
- product:      product team, SLA 24h, medium priority
                Covers: feature requests, how-to questions, documentation gaps
- complaint:    customer success, SLA 1h, urgent
                Covers: angry customers, escalation threats, repeated unresolved issues
- general:      first-line,   SLA 48h, low priority
                Covers: general inquiries, account queries, other
$$;

CREATE FUNCTION response_tone_guide(tone TEXT DEFAULT 'professional')
RETURNS TEXT
AS $$
SELECT CASE tone
  WHEN 'empathetic' THEN 'Warm and understanding. Acknowledge frustration. Use "I understand" and "I am sorry". Avoid jargon. Refer to the customer by first name.'
  WHEN 'formal'     THEN 'Professional and precise. Use formal language. Reference ticket numbers and order IDs explicitly. State next steps clearly.'
  WHEN 'friendly'   THEN 'Conversational and positive. Use the customer first name. End with an enthusiastic offer to help further.'
  ELSE                   'Professional but approachable. Concise. Action-oriented. Reference specific order details when available. No excessive apologies.'
END
$$;

CREATE FUNCTION order_context_prompt(order_context TEXT)
RETURNS TEXT
AS $$
SELECT CASE
  WHEN order_context = 'No order number found in this ticket.'
    THEN 'No order number was referenced — respond based on the ticket text alone.'
  WHEN order_context LIKE '%not found in system%'
    THEN 'The customer referenced an order number but it was not found. Ask them to confirm the order ID.'
  ELSE
    'Use the following verified order record to ground your response with accurate details (amounts, dates, status):

' || order_context
END
$$;

    GENERATE detect_urgency(@ticket, @classification_json) INTO @urgency_score

            GENERATE draft_response(
                @ticket,
                @classification_json,
                @details_json,
                @order_context,
                @product,
                response_tone_guide(@tone)
            ) INTO @drafted_response

            -- Step 8: Quality-check the draft (0-10); revise once if score < 6
            GENERATE check_response_quality(@drafted_response, @ticket, @order_context) INTO @quality_score

                    GENERATE improve_response(@drafted_response, @ticket, @order_context) INTO @drafted_response
                    COMMIT @drafted_response WITH status = 'drafted_revised'
                ELSE
                    COMMIT @drafted_response WITH status = 'drafted', quality = @quality_score
            END
    END

EXCEPTION
    WHEN GenerationError THEN
        GENERATE fallback_response(@ticket, @product) INTO @drafted_response
        COMMIT @drafted_response WITH status = 'fallback'
END

CREATE FUNCTION language_conventions(language TEXT)
RETURNS TEXT
AS $$
SELECT CASE language
  WHEN 'Python' THEN 'PEP 8 style. Type hints required. Docstring in Google format. No bare except.'
  WHEN 'Go'     THEN 'Standard Go conventions. Return (value, error). Table-driven tests. No panics in library code.'
  WHEN 'TypeScript' THEN 'Strict mode. Explicit return types. JSDoc comments. No any type.'
  WHEN 'Rust'   THEN 'Idiomatic Rust. Result<T, E> for fallible ops. No unwrap() in library code. Doc comments.'

CREATE FUNCTION test_framework_guide(language TEXT, framework TEXT DEFAULT 'default')
RETURNS TEXT
AS $$

  WHEN 'Python:pytest'   THEN 'pytest. Use fixtures. Parametrize for multiple inputs. Assert with descriptive messages.'

  WHEN 'TypeScript:jest' THEN 'Jest. describe/it blocks. Mocks via jest.fn(). Async tests with async/await.'
  ELSE                        'Standard test framework for ' || language || '. Cover: happy path, edge cases, error cases.'
END
$$;

-- Recipe 34: Progressive Summarizer
-- Layered summary: sentence → paragraph → page.
-- Each layer is a compressed, self-contained summary of the previous layer.
--
-- Usage:
--   spl run cookbook/34_progressive_summary/progressive_summary.spl --adapter ollama -m gemma3 \
--       text="$(cat long_article.txt)"
--
--   spl run cookbook/34_progressive_summary/progressive_summary.spl --adapter ollama \
--       text="$(cat research_paper.txt)" audience="executive" layers=3

CREATE FUNCTION summary_constraints(layer TEXT)
RETURNS TEXT
AS $$
SELECT CASE layer
  WHEN 'sentence'  THEN 'One sentence only. Maximum 25 words. Capture the single most important idea.'
  WHEN 'paragraph' THEN '3-5 sentences. Cover main points and key supporting evidence. No examples or details.'
  WHEN 'page'      THEN '2-3 paragraphs. Include context, main argument, evidence, and implications.'
  ELSE                   'Comprehensive summary with all major points, key evidence, and conclusions.'
END
$$;

WORKFLOW progressive_summarizer
    INPUT:
        @text TEXT,
        @audience TEXT DEFAULT 'general',
        @layers INT DEFAULT 3
    OUTPUT: @summary_package TEXT
DO
    LOGGING f'Progressive summary | audience={@audience} layers={@layers}' LEVEL INFO

    -- Layer 1: One-sentence summary
    GENERATE summarize(
        @text,
        summary_constraints('sentence'),
        @audience
    ) INTO @sentence_summary
    LOGGING f'Layer 1 done | sentence summary ready' LEVEL DEBUG

    -- Layer 2: Paragraph summary (built on layer 1 for coherence)
    GENERATE expand_summary(
        @text,
        @sentence_summary,
        summary_constraints('paragraph'),
        @audience
    ) INTO @paragraph_summary
    LOGGING f'Layer 2 done | paragraph summary ready' LEVEL DEBUG

    -- Layer 3: Page-length summary (built on layer 2)
    EVALUATE @layers
        WHEN >= 3 THEN
            GENERATE expand_summary(
                @text,
                @paragraph_summary,
                summary_constraints('page'),
                @audience
            ) INTO @page_summary
            LOGGING 'Layer 3 done | page summary ready' LEVEL DEBUG
        ELSE
            @page_summary := ''
    END

    -- Quality check: does each layer faithfully represent the original?
    GENERATE verify_summary_fidelity(@text, @sentence_summary, @paragraph_summary) INTO @fidelity_score
    LOGGING f'Fidelity score: {@fidelity_score}' LEVEL INFO

    -- Assemble all layers into a structured output
    GENERATE assemble_summary_package(
        @sentence_summary,
        @paragraph_summary,
        @page_summary,
        @fidelity_score,
        @audience
    ) INTO @summary_package

    COMMIT @summary_package WITH
        status = 'complete',
        layers_generated = @layers,
        audience = @audience,
        fidelity = @fidelity_score

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Text too long — chunk first then summarize each chunk
        GENERATE chunk_and_summarize(@text, summary_constraints('paragraph')) INTO @paragraph_summary
        GENERATE summarize(@paragraph_summary, summary_constraints('sentence'), @audience) INTO @sentence_summary
        GENERATE assemble_summary_package(@sentence_summary, @paragraph_summary, '', 'n/a', @audience) INTO @summary_package

-- Generate hypothesis → design test → evaluate evidence → conclude.
-- Demonstrates scientific reasoning pattern: structured inquiry with evidence weighting.
--
-- Usage:
--   spl run cookbook/35_hypothesis_tester/hypothesis.spl --adapter ollama -m gemma3 \
--       observation="Remote teams show lower productivity in the first month after joining"

--       observation="Users who receive onboarding emails churn 30% less" \

CREATE FUNCTION hypothesis_framework()
RETURNS TEXT
AS $$
Scientific hypothesis format:
- Null hypothesis (H0): The effect does not exist / is due to chance
- Alternative hypothesis (H1): The proposed causal or correlational claim
- Key variables: independent, dependent, confounding
- Testability: How would we falsify H0?
$$;

CREATE FUNCTION evidence_schema()
RETURNS JSON
AS $$
{
  "type": "object",
  "properties": {
    "supports_h1":   {"type": "array", "items": {"type": "string"}},
    "supports_h0":   {"type": "array", "items": {"type": "string"}},
    "inconclusive":  {"type": "array", "items": {"type": "string"}},
    "confidence":    {"type": "number", "minimum": 0, "maximum": 1},
    "verdict":       {"type": "string", "enum": ["reject_h0", "fail_to_reject_h0", "inconclusive"]},
    "caveats":       {"type": "array", "items": {"type": "string"}}
  },
  "required": ["confidence", "verdict"]
}
$$;

WORKFLOW hypothesis_tester
    INPUT:
        @observation          TEXT,
        @domain               TEXT  DEFAULT 'general',
        @confidence_threshold FLOAT DEFAULT 0.7
    OUTPUT: @conclusion TEXT
DO
    LOGGING f'Hypothesis tester | domain={@domain} threshold={@confidence_threshold}' LEVEL INFO

    -- Step 1: Formulate hypotheses
    GENERATE formulate_hypotheses(
        @observation,
        @domain,
        hypothesis_framework()
    ) INTO @hypotheses
    LOGGING 'Hypotheses formulated' LEVEL DEBUG

    -- Step 2: Design a test plan
    GENERATE design_test(
        @hypotheses,
        @domain
    ) INTO @test_plan
    LOGGING 'Test plan designed' LEVEL DEBUG

    -- Step 3: Gather and evaluate available evidence
    GENERATE evaluate_evidence(
        @observation,
        @hypotheses,
        @test_plan,
        evidence_schema()
    ) INTO @evidence_json
    LOGGING 'Evidence evaluated' LEVEL DEBUG

    -- Step 4: Extract confidence score
    GENERATE extract_confidence(@evidence_json) INTO @confidence
    LOGGING f'Confidence score: {@confidence} | threshold={@confidence_threshold}' LEVEL INFO

    -- Step 5: Draw conclusion based on evidence strength
    EVALUATE @confidence
        WHEN >= @confidence_threshold THEN
            GENERATE write_conclusion(
                @hypotheses, @evidence_json, @test_plan, 'confident'
            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'concluded',
                verdict = 'h1_supported',
                confidence = @confidence
        WHEN >= 0.4 THEN

            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'inconclusive',
                verdict = 'needs_more_data',
                confidence = @confidence
        ELSE
            GENERATE write_conclusion(
                @hypotheses, @evidence_json, @test_plan, 'refuted'
            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'concluded',
                verdict = 'h0_not_rejected',
                confidence = @confidence
    END

EXCEPTION
    WHEN GenerationError THEN
        COMMIT @hypotheses WITH status = 'hypotheses_only', reason = 'evidence_evaluation_failed'

-- Recipe 36: Tool-Use / Function-Call
-- Demonstrates registering Python functions as CALL-able tools in SPL 2.0.
--
-- Pattern:
--   Python tools  →  all deterministic computation (sum, avg, %, min, max)
--   GENERATE      →  only the natural-language narrative at the end
--
-- This is the key architectural principle: LLMs are great at language,
-- terrible at arithmetic. Keep them in their lane.
--
-- Usage:
--   spl run cookbook/36_tool_use/tool_use.spl \
--       --adapter ollama \
--       --tools cookbook/36_tool_use/tools.py \
--       sales="1200,1450,1380,1600,1750,1900" \
--       prev_total="7800" \
--       period="H1 2025"
--
-- Tool dispatch priority in CALL:
--   1. Python tool (registered via --tools)   ← used here for all math
--   2. SPL PROCEDURE
--   3. LLM fallback (last resort — avoid)

WORKFLOW sales_analysis
  INPUT:
    @sales     TEXT DEFAULT '1200,1450,1380,1600,1750,1900',
    @prev_total TEXT DEFAULT '7800',
    @period    TEXT DEFAULT 'H1 2025'
  OUTPUT: @report TEXT
DO
  -- All math delegated to Python tools — deterministic, instant, zero tokens
  CALL sum_values(@sales)            INTO @total
  CALL average_values(@sales)        INTO @avg
  CALL min_value(@sales)             INTO @low
  CALL max_value(@sales)             INTO @high
  CALL percentage_change(@prev_total, @total) INTO @growth
  CALL format_currency(@total)       INTO @total_fmt
  CALL format_currency(@avg)         INTO @avg_fmt

  -- Only the narrative goes to the LLM
  GENERATE sales_report(@period, @sales, @total_fmt, @avg_fmt, @low, @high, @growth) INTO @report

  COMMIT @report WITH status = 'complete'

EXCEPTION
  WHEN OTHERS THEN
    GENERATE error_summary(@period) INTO @report
    COMMIT @report WITH status = 'error'
END

-- Recipe 37: Headline News Aggregator
-- Generate, expand, and evaluate a daily news digest for a given topic area.
-- Uses claude_cli adapter with Opus model for broad knowledge synthesis.
--
-- Pattern:
--   generate_headlines(topic) → N headlines
--     → expand each headline into a brief summary
--     → evaluate coverage and balance
--     → refine if coverage gaps detected
--     → commit structured daily digest
--
-- Usage:
--   spl run cookbook/37_headline_news/headline_news.spl --adapter claude_cli -m claude-opus-4-6 \
--       topic="artificial intelligence"
--
--   spl run cookbook/37_headline_news/headline_news.spl --adapter claude_cli -m claude-opus-4-6 \
--       topic="renewable energy" format="executive brief" max_headlines=5
--
--   spl run cookbook/37_headline_news/headline_news.spl --adapter claude_cli -m claude-opus-4-6 \
--       topic="quantum computing" format="bullet points" perspective="global"

CREATE FUNCTION news_format_guide(format TEXT DEFAULT 'structured')
RETURNS TEXT
AS $$
SELECT CASE format
  WHEN 'executive brief'  THEN 'Concise executive summary. Lead with the most significant development. 2-3 sentences per item. Close with a one-line strategic implication.'
  WHEN 'bullet points'    THEN 'Bulleted list. One headline per bullet. Sub-bullets for key facts. No more than 3 sub-bullets per headline.'
  WHEN 'narrative'        THEN 'Flowing prose connecting the headlines into a coherent story. Identify themes and through-lines. 3-4 paragraphs.'
  ELSE                         'Structured digest. Each headline followed by a 2-3 sentence summary. Group related items. End with a brief synthesis.'
END
$$;

CREATE FUNCTION perspective_guide(perspective TEXT DEFAULT 'balanced')
RETURNS TEXT
AS $$
SELECT CASE perspective
  WHEN 'technical'   THEN 'Focus on technical developments, research breakthroughs, and engineering milestones. Prioritize substance over narrative.'
  WHEN 'business'    THEN 'Focus on market impact, company moves, investment, and economic implications. Prioritize commercial significance.'
  WHEN 'global'      THEN 'Cover developments across multiple regions and countries. Note geographic variation in trends. Avoid US-centric framing.'
  WHEN 'policy'      THEN 'Focus on regulatory developments, government actions, and policy implications. Note geopolitical dimensions.'
  ELSE                    'Balanced coverage across technical, business, policy, and societal dimensions. Represent multiple viewpoints.'
END
$$;

CREATE FUNCTION generate_headlines(topic TEXT, max_headlines TEXT, date TEXT, perspective TEXT) RETURNS TEXT
AS $$
You are a news editor with broad knowledge. Generate the top {max_headlines} headlines about "{topic}" as of {date}.

Coverage perspective: {perspective}

Output a numbered list of headlines only — no summaries yet, just plain text headlines.
1. First headline here
2. Second headline here
$$;

CREATE FUNCTION expand_headlines(headlines TEXT, topic TEXT, perspective TEXT) RETURNS TEXT
AS $$
You are a news analyst. Expand each headline in this list with a 2-3 sentence factual summary.

Topic area: {topic}
Perspective lens: {perspective}

Headlines to expand:
{headlines}

For each headline, write:
[N]. [Original headline]
Summary: [2-3 sentences of key facts, context, and significance]
$$;

CREATE FUNCTION evaluate_coverage(expanded TEXT, topic TEXT, perspective TEXT) RETURNS TEXT
AS $$
You are a news editor evaluating coverage quality. Assess how well these summaries cover the major angles for "{topic}".

Perspective: {perspective}

Summaries:
{expanded}

Reply with ONLY a decimal score between 0.0 and 1.0 representing coverage completeness.
- 1.0 = all major angles covered
- 0.75 = most angles covered, minor gaps
- 0.5 = significant gaps
- 0.25 = very incomplete

Score only, no explanation.
$$;

CREATE FUNCTION fill_coverage_gaps(expanded TEXT, topic TEXT, coverage_score TEXT, perspective TEXT) RETURNS TEXT
AS $$
You are a news editor filling coverage gaps. The current digest scored {coverage_score} on coverage.

Topic: {topic}
Perspective: {perspective}

Current summaries:
{expanded}

Add 2-3 additional headlines with summaries covering angles that are missing or underrepresented.
Return the FULL list (original + additions) in the same format.
$$;

CREATE FUNCTION format_digest(expanded TEXT, topic TEXT, date TEXT, format_guide TEXT) RETURNS TEXT
AS $$
You are a news editor formatting a daily digest.

Topic: {topic}
Date: {date}
Format instructions: {format_guide}

Content to format:
{expanded}

Format the content according to the format instructions. Include a header with topic and date.
$$;

WORKFLOW headline_news
    INPUT:
        @topic         TEXT,
        @date          TEXT    DEFAULT 'today',
        @max_headlines INTEGER DEFAULT 7,
        @format        TEXT    DEFAULT 'structured',
        @perspective   TEXT    DEFAULT 'balanced'
    OUTPUT: @digest TEXT
DO
    LOGGING f'Headline news | topic={@topic} max={@max_headlines} perspective={@perspective}' LEVEL INFO

    -- Step 1: Generate top headlines for the topic
    GENERATE generate_headlines(
        @topic,
        @max_headlines,
        @date,
        perspective_guide(@perspective)
    ) INTO @headlines
    LOGGING f'Headlines generated: {@max_headlines} items' LEVEL DEBUG

    -- Step 2: Expand each headline into a brief summary with key facts
    GENERATE expand_headlines(
        @headlines,
        @topic,
        perspective_guide(@perspective)
    ) INTO @expanded
    LOGGING 'Headlines expanded' LEVEL DEBUG

    -- Step 3: Evaluate coverage — are major angles represented?
    GENERATE evaluate_coverage(
        @expanded,
        @topic,
        perspective_guide(@perspective)
    ) INTO @coverage_score
    LOGGING f'Coverage score: {@coverage_score}' LEVEL INFO

    -- Step 4: Refine if coverage gaps detected
    EVALUATE @coverage_score
        WHEN > 0.75 THEN
            -- Coverage is solid — format and commit
            GENERATE format_digest(
                @expanded,
                @topic,
                @date,
                news_format_guide(@format)
            ) INTO @digest
            COMMIT @digest WITH status = 'complete', coverage = @coverage_score

        ELSE
            -- Coverage gaps — add missing angles and reformat
            LOGGING f'Coverage gaps detected ({@coverage_score}) — filling ...' LEVEL WARN
            GENERATE fill_coverage_gaps(
                @expanded,
                @topic,
                @coverage_score,
                perspective_guide(@perspective)
            ) INTO @expanded

            GENERATE format_digest(
                @expanded,
                @topic,
                @date,
                news_format_guide(@format)
            ) INTO @digest
            COMMIT @digest WITH status = 'refined', coverage = @coverage_score
    END

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Too many headlines — trim and commit best effort
        GENERATE format_digest(
            @headlines,
            @topic,
            @date,
            news_format_guide(@format)
        ) INTO @digest
        COMMIT @digest WITH status = 'partial'

    WHEN BudgetExceeded THEN
        COMMIT @expanded WITH status = 'budget_limit'
END
