CREATE FUNCTION sentiment_schema()
RETURNS JSON
AS $$
{
  "type": "array",
  "description": "One entry per input item, in the same order as the input list.",
  "items": {
    "type": "object",

      "score":      {"type": "number", "minimum": -1, "maximum": 1,
                     "description": "Sentiment score: -1 most negative, +1 most positive"},
      "confidence": {"type": "number", "minimum": 0, "maximum": 1,
                     "description": "Model confidence in the label"},
      "emotions":   {"type": "array", "items": {"type": "string"},
                     "description": "Detected emotions e.g. joy, anger, frustration, surprise"},
      "key_phrases":{"type": "array", "items": {"type": "string"},
                     "description": "Most sentiment-bearing phrases from the text"}
    },
    "required": ["item", "label", "score", "confidence"]
  }
}
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW sentiment_pipeline
    INPUT:  @filename  TEXT DEFAULT '',
            @items     TEXT DEFAULT '',
            @delimiter TEXT DEFAULT '\n',
            @domain    TEXT DEFAULT 'general'
    OUTPUT: @report TEXT
DO
    LOGGING f'Sentiment pipeline | domain={@domain} filename={@filename}' LEVEL INFO

    -- Step 1: Load items from file if filename provided (deterministic — no LLM)
    CALL load_items(@filename) INTO @file_content

    -- Step 2: Split into a clean JSON array (deterministic — no LLM)
    CALL split_items(@file_content, @items, @delimiter) INTO @item_list
    LOGGING f'Items loaded and split | list={@item_list}' LEVEL DEBUG

    -- Step 3: Batch sentiment — one structured result per item (LLM)
    LOGGING 'Running batch sentiment ...' LEVEL INFO
    GENERATE batch_sentiment(@item_list, sentiment_schema(), @domain) INTO @sentiment_results

    -- Step 4: Compute aggregate statistics (deterministic — no LLM)
    CALL compute_stats(@sentiment_results) INTO @stats_json
    LOGGING f'Stats computed: {@stats_json}' LEVEL DEBUG

    -- Step 5: Identify extreme and notable items (deterministic — no LLM)
    CALL find_extremes(@sentiment_results) INTO @extremes

    -- Step 6: Generate trend narrative from stats and extremes (LLM)
    GENERATE summarize_sentiment_trends(
        @stats_json,
        @extremes,
        @domain
    ) INTO @trend_summary
    LOGGING 'Trend narrative ready' LEVEL DEBUG

    -- Step 7: Assemble the full report (LLM)
    GENERATE assemble_sentiment_report(
        @sentiment_results,
        @stats_json,
        @trend_summary,
        @extremes,
        @domain
    ) INTO @report

    LOGGING f'Sentiment report complete | domain={@domain}' LEVEL INFO
    COMMIT @report WITH status = 'complete', domain = @domain

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Too many items — skip per-item detail, return stats + trend only
        CALL compute_stats(@sentiment_results) INTO @stats_json
        CALL find_extremes(@sentiment_results) INTO @extremes
        GENERATE summarize_sentiment_trends(@stats_json, @extremes, @domain) INTO @trend_summary
        SET @report = @trend_summary
        COMMIT @report WITH status = 'stats_only', reason = 'context_length_exceeded'
END

def summarize_sentiment_trends(product_reviews):
    """
    Summarize trends in sentiment analysis for a list of product reviews.

    Args:
        product_reviews (dict): A dictionary mapping each review to its corresponding sentiment score.
            Each value is another dictionary containing the sentiment score, including 'compound' and 'positive', 'negative'.

    Returns:
        dict: A dictionary summarizing the trends in sentiment analysis.
            The dictionary contains keys for 'overall_trend', 'positive_trend', 'negative_trend'.
    """

    # Check if the input is valid
    if not isinstance(product_reviews, dict):
        return {"error": "Invalid input type. Expected a dictionary."}

    # Initialize variables to store the results
    overall_score = 0
    positive_count = 0
    negative_count = 0

    # Iterate over each review in the product reviews
    for review in product_reviews.values():
        # Extract the compound and polarity scores from the sentiment analysis result
        score = review['compound']
        polarity = review['positive'] - review['negative']

        # Update the overall score
        overall_score += score

        # Update the positive and negative count
        if score > 0:
            positive_count += 1
        elif score < 0:
            negative_count += 1

    # Calculate the average compound score
    avg_compound = overall_score / len(product_reviews)

    # Determine the trend based on the polarity scores
    if polarity > 0:
        trend = 'positive'
    elif polarity < 0:
        trend = 'negative'
    else:
        trend = 'neutral'

    # Summarize the trends in sentiment analysis
    summary = {
        "overall_trend": trend,
        "positive_trend": positive_count / len(product_reviews) if product_reviews else 0,
        "negative_trend": negative_count / len(product_reviews) if product_reviews else 0,
        "avg_compound": avg_compound
    }

    return summary

import nltk
from nltk.sentiment.vader import SentimentIntensityAnalyzer

def batch_sentiment(input_list):
    """
    Calculate sentiment for a list of product reviews.

    Args:
        input_list (list): A list of strings representing product reviews.
            Each review is separated by '|' and contains the text and 
            sentiment label.

    Returns:
        dict: A dictionary mapping each review to its corresponding sentiment score.
    """

    # Initialize NLTK data needed for sentiment analysis
    nltk.download('vader_lexicon')

    # Create a SentimentIntensityAnalyzer object
    sia = SentimentIntensityAnalyzer()

    # Initialize an empty dictionary to store the results
    results = {}

    # Iterate over each review in the input list
    for review in input_list:
        # Split the review into text and sentiment label
        text, _ = review.split("|")

        # Get the sentiment score of the review
        sentiment_score = sia.polarity_scores(text)

        # Store the result in the dictionary
        results[review] = sentiment_score

    return results
