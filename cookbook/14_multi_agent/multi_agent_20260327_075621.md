```spl
-- Recipe 14: Multi-Agent Collaboration
-- Specialized agents (Researcher, Analyst, Writer) collaborate on a report.
-- Each agent has a distinct role and passes results to the next.
-- Demonstrates procedural decomposition, CALL-based delegation, and role specialization.
--
-- Usage:
--   spl run cookbook/14_multi_agent/multi_agent.spl --adapter ollama -m gemma3 topic="Impact of AI on healthcare"
--   spl run cookbook/14_multi_agent/multi_agent.spl --adapter ollama topic="Future of renewable energy"
--   spl run cookbook/14_multi_agent/multi_agent.spl --adapter ollama topic="State of quantum computing in 2025"
--
-- Logs written to @log_dir (default: cookbook/14_multi_agent/logs):
--   research.md   — facts and key themes from Researcher agent
--   analysis.md   — trends, risks, and opportunities from Analyst agent
--   report.md     — final polished report from Writer agent

-- Agent 1: Researcher — gathers key facts and sources
PROCEDURE researcher(topic TEXT) RETURNS TEXT
DO
    GENERATE research_facts(topic) INTO @facts
    GENERATE identify_key_themes(@facts) INTO @themes
    @result := @facts || '\n\nKey Themes:\n' || @themes
    COMMIT @result
END

-- Agent 2: Analyst — identifies trends, risks, and opportunities
PROCEDURE analyst(research TEXT, topic TEXT) RETURNS TEXT
DO
    GENERATE analyze_trends(research) INTO @trends
    GENERATE assess_risks(research, topic) INTO @risks
    GENERATE find_opportunities(research, topic) INTO @opportunities
    @result := 'Trends:\n' || @trends || '\n\nRisks:\n' || @risks || '\n\nOpportunities:\n' || @opportunities
    COMMIT @result
END

-- Agent 3: Writer — produces polished final report
PROCEDURE writer(research TEXT, analysis TEXT, topic TEXT) RETURNS TEXT
DO
    GENERATE draft_report(topic, research, analysis) INTO @draft
    GENERATE critique(@draft) INTO @feedback
    GENERATE revise_report(@draft, @feedback) INTO @final
    COMMIT @final
END

-- Orchestrator workflow
WORKFLOW multi_agent_report
    INPUT: 
        @topic TEXT,
        @log_dir TEXT DEFAULT 'cookbook/14_multi_agent/logs'
    OUTPUT: @report TEXT
DO
    LOGGING f'Multi-agent report | topic={@topic}' LEVEL INFO

    -- Delegate to specialized agents
    LOGGING 'Agent 1: Researcher ...' LEVEL DEBUG
    CALL researcher(@topic) INTO @research
    LOGGING f'Research:\n{@research}' LEVEL DEBUG
    CALL write_file(f'{@log_dir}/research.md', @research) INTO @_
    LOGGING 'Agent 2: Analyst ...' LEVEL DEBUG
    CALL analyst(@research, @topic) INTO @analysis
    LOGGING f'Analysis:\n{@analysis}' LEVEL DEBUG
    CALL write_file(f'{@log_dir}/analysis.md', @analysis) INTO @_
    LOGGING 'Agent 3: Writer ...' LEVEL DEBUG
    CALL writer(@research, @analysis, @topic) INTO @report
    LOGGING f'Report:\n{@report}' LEVEL DEBUG
    CALL write_file(f'{@log_dir}/report.md', @report) INTO @_

    LOGGING 'Report complete' LEVEL INFO
    COMMIT @report WITH status = 'complete'

EXCEPTION
    WHEN BudgetExceeded THEN
        COMMIT @research WITH status = 'partial_research_only'
    WHEN HallucinationDetected THEN
        RETRY WITH temperature = 0.1 LIMIT 3
END
```
