}
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW meeting_to_actions
    INPUT:  @filename      TEXT DEFAULT '',
            @transcript    TEXT DEFAULT '',
            @output_format TEXT DEFAULT 'json'
    OUTPUT: @output TEXT
DO
    LOGGING f'Meeting to actions | format={@output_format} filename={@filename}' LEVEL INFO

    -- Step 1: Load transcript from file if filename provided (deterministic — no LLM)
    CALL load_transcript(@filename) INTO @file_content

    -- Step 2: Extract speaker names for grounding the LLM (deterministic — no LLM)
    CALL extract_speakers(@file_content) INTO @speakers_from_file
    CALL extract_speakers(@transcript)   INTO @speakers_from_inline
    LOGGING f'Speakers from file: {@speakers_from_file}' LEVEL DEBUG

    -- Step 3: Normalise transcript — merge sources, clean formatting, identify speakers
    GENERATE normalize_transcript(
        @file_content,
        @transcript,
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @clean_transcript
    LOGGING 'Transcript normalised' LEVEL DEBUG

    -- Step 4: Extract structured action items grounded by schema and known speakers
    GENERATE extract_actions(
        @clean_transcript,
        action_item_schema(),
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @structured_json
    LOGGING 'Action items extracted' LEVEL INFO

    -- Step 5: Normalise relative due dates to ISO-8601 (deterministic — no LLM)
    CALL normalize_dates(@structured_json) INTO @structured_json

    -- Step 6: Validate ownership — flag unowned high-priority items (deterministic — no LLM)
    CALL validate_ownership(@structured_json) INTO @validation_notes

    -- Step 7: Format and commit output
    EVALUATE @output_format
        WHEN = 'markdown' THEN
            GENERATE format_as_markdown(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'markdown'
        WHEN = 'email' THEN
            GENERATE format_as_email(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'email'
        ELSE
            SET @output = @structured_json
            COMMIT @output WITH status = 'complete', format = 'json'
    END

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Long transcript: summarise first, then extract from the summary
        GENERATE summarize_transcript(@clean_transcript) INTO @summary
        GENERATE extract_actions(@summary, action_item_schema(), @speakers_from_file, @speakers_from_inline) INTO @structured_json
        CALL normalize_dates(@structured_json) INTO @structured_json
        CALL validate_ownership(@structured_json) INTO @validation_notes
        SET @output = @structured_json
        COMMIT @output WITH status = 'complete_chunked'
END

```python
import re
from collections import defaultdict

def extract_actions(transcript):
    """
    Extract actions from a transcript using regular expressions.

    Args:
        transcript (str): The transcript to extract actions from.

    Returns:
        str: A formatted string containing the extracted actions.
    """

    # Initialize a dictionary to store the speakers and their actions
    speaker_dict = defaultdict(list)

    # Split the transcript into lines
    lines = transcript.split('\n')

    # Iterate over each line in the transcript
    for i, line in enumerate(lines):
        # Use regular expression to extract the action from the line
        match = re.search(r'([a-zA-Z]+): (.*)', line)
        
        if match:
            # If an action is found, add it to their list in the dictionary
            speaker_dict[match.group(1)].append(match.group(2).strip())
        else:
            # If no speaker or action is found, ignore this line

    # Return the extracted actions as a string
    return '\n'.join(f'{speaker}: {", ".join(action)}' for speaker, action in speaker_dict.items())
