-- ── Schema (zero LLM cost — injected as grounding context) ─────────────────

CREATE FUNCTION action_item_schema()
RETURNS JSON
AS $$
{
  "type": "object",
  "properties": {
    "meeting_title":   {"type": "string"},
    "meeting_date":    {"type": "string"},
    "attendees":       {"type": "array", "items": {"type": "string"}},
    "meeting_summary": {"type": "string"},
    "decisions": {
      "type": "array",
      "items": {"type": "string"}
    },
    "action_items": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "task":     {"type": "string"},
          "owner":    {"type": "string", "description": "Person responsible. Use exact speaker name or empty string if unassigned."},
          "due_date": {"type": "string", "description": "Due date or relative phrase e.g. Friday, next Wednesday, end of sprint"},
          "priority": {"type": "string", "enum": ["high", "medium", "low"]},
          "context":  {"type": "string", "description": "Why this task matters or what triggered it"}
        },
        "required": ["task", "priority"]
      }
    },
    "open_questions": {
      "type": "array",
      "items": {"type": "string"}
    },
    "next_meeting": {"type": "string"}
  },
  "required": ["meeting_summary", "action_items"]
}
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW meeting_to_actions
    INPUT:  @filename      TEXT DEFAULT '',
            @transcript    TEXT DEFAULT '',
            @output_format TEXT DEFAULT 'json'
    OUTPUT: @output TEXT
DO
    LOGGING f'Meeting to actions | format={@output_format} filename={@filename}' LEVEL INFO

    -- Step 1: Load transcript from file if filename provided (deterministic — no LLM)
    CALL load_transcript(@filename) INTO @file_content

    -- Step 2: Extract speaker names for grounding the LLM (deterministic — no LLM)
    CALL extract_speakers(@file_content) INTO @speakers_from_file
    CALL extract_speakers(@transcript)   INTO @speakers_from_inline
    LOGGING f'Speakers from file: {@speakers_from_file}' LEVEL DEBUG

    -- Step 3: Normalise transcript — merge sources, clean formatting, identify speakers
    GENERATE normalize_transcript(
        @file_content,
        @transcript,
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @clean_transcript
    LOGGING 'Transcript normalised' LEVEL DEBUG

    -- Step 4: Extract structured action items grounded by schema and known speakers
    GENERATE extract_actions(
        @clean_transcript,
        action_item_schema(),
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @structured_json
    LOGGING 'Action items extracted' LEVEL INFO

    -- Step 5: Normalise relative due dates to ISO-8601 (deterministic — no LLM)
    CALL normalize_dates(@structured_json) INTO @structured_json

    -- Step 6: Validate ownership — flag unowned high-priority items (deterministic — no LLM)
    CALL validate_ownership(@structured_json) INTO @validation_notes

    -- Step 7: Format and commit output
    EVALUATE @output_format
        WHEN = 'markdown' THEN
            GENERATE format_as_markdown(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'markdown'
        WHEN = 'email' THEN
            GENERATE format_as_email(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'email'
        ELSE
            SET @output = @structured_json
            COMMIT @output WITH status = 'complete', format = 'json'
    END

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Long transcript: summarise first, then extract from the summary
        GENERATE summarize_transcript(@clean_transcript) INTO @summary
        GENERATE extract_actions(@summary, action_item_schema(), @speakers_from_file, @speakers_from_inline) INTO @structured_json
        CALL normalize_dates(@structured_json) INTO @structured_json
        CALL validate_ownership(@structured_json) INTO @validation_notes
        SET @output = @structured_json
        COMMIT @output WITH status = 'complete_chunked'
END

import re
from collections import defaultdict

def normalize_transcript(transcript):
    # Create a dictionary to store speakers, messages, and actions
    transcript_data = defaultdict(lambda: {'messages': [], 'actions': []})
    
    # Split the transcript into lines
    lines = transcript.splitlines()
    
    # Iterate over each line in the transcript
    for line in lines:
        # Use regular expression to extract the speaker's name from the line
        match = re.search(r'([a-zA-Z]+):', line)
        
        if match:
            speaker = match.group(1).strip()
            
            # If a speaker is found, add their message to the list of messages for that speaker
            transcript_data[speaker]['messages'].append(line.strip())
        else:
            # Use regular expression to extract action items from the line
            action_match = re.search(r'([a-zA-Z]+): ([^:]+)', line)
            
            if action_match:
                speaker, action = action_match.groups()
                speaker = speaker.strip()
                
                # If an action is found, add it to the list of actions for that speaker
                transcript_data[speaker]['actions'].append((action, line.strip()))
    
    # Join all messages for each speaker into a single string
    normalized_transcript = '\n'.join(f'{speaker}: {"".join(messages)}' for speaker, messages in transcript_data.items())
    
    # Format actions as requested
    formatted_actions = []
    for speaker, actions in transcript_data.items():
        if actions['actions']:
            formatted_actions.append({

                'actions': [f"{action[0]}: {action[1]}" for action in actions['actions']]
            })
    
    # Append formatted actions to the normalized transcript
    if formatted_actions:
        return f'{normalized_transcript}\nActions:\n{", ".join(str(action) for action in formatted_actions)}'
