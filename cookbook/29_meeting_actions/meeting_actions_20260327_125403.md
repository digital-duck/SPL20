-- Transcript in, structured TODO list + owners out.
-- Demonstrates multi-step extraction + structured output from long-form text,
-- with deterministic tool calls for file loading, speaker extraction,
-- date normalisation, and ownership validation.

-- Usage (load from file):

-- Sample transcripts in transcripts/:
--   sprint_planning.txt   Full sprint planning with assignments and decisions
--   design_review.txt     API architecture review with open questions
--   standup.txt           Short daily standup with blockers

-- ── Schema (zero LLM cost — injected as grounding context) ─────────────────

CREATE FUNCTION action_item_schema()
RETURNS JSON
AS $$
{
  "type": "object",
  "properties": {
    "meeting_title":   {"type": "string"},
    "meeting_date":    {"type": "string"},
    "attendees":       {"type": "array", "items": {"type": "string"}},
    "meeting_summary": {"type": "string"},
    "decisions": {
      "type": "array",
      "items": {"type": "string"}
    },
    "action_items": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "task":     {"type": "string"},
          "owner":    {"type": "string", "description": "Person responsible. Use exact speaker name or empty string if unassigned."},
          "due_date": {"type": "string", "description": "Due date or relative phrase e.g. Friday, next Wednesday, end of sprint"},
          "priority": {"type": "string", "enum": ["high", "medium", "low"]},
          "context":  {"type": "string", "description": "Why this task matters or what triggered it"}
        },
        "required": ["task", "priority"]
      }
    },
    "open_questions": {
      "type": "array",
      "items": {"type": "string"}
    },
    "next_meeting": {"type": "string"}
  },
  "required": ["meeting_summary", "action_items"]
}
$$;

-- ── Main workflow ───────────────────────────────────────────────────────────

WORKFLOW meeting_to_actions
    INPUT:  @filename      TEXT DEFAULT '',
            @transcript    TEXT DEFAULT '',
            @output_format TEXT DEFAULT 'json'
    OUTPUT: @output TEXT
DO
    LOGGING f'Meeting to actions | format={@output_format} filename={@filename}' LEVEL INFO

    -- Step 1: Load transcript from file if filename provided (deterministic — no LLM)
    CALL load_transcript(@filename) INTO @file_content

    -- Step 2: Extract speaker names for grounding the LLM (deterministic — no LLM)
    CALL extract_speakers(@file_content) INTO @speakers_from_file
    CALL extract_speakers(@transcript)   INTO @speakers_from_inline
    LOGGING f'Speakers from file: {@speakers_from_file}' LEVEL DEBUG

    -- Step 3: Normalise transcript — merge sources, clean formatting, identify speakers
    GENERATE normalize_transcript(
        @file_content,
        @transcript,
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @clean_transcript
    LOGGING 'Transcript normalised' LEVEL DEBUG

    -- Step 4: Extract structured action items grounded by schema and known speakers
    GENERATE extract_actions(
        @clean_transcript,
        action_item_schema(),
        @speakers_from_file,
        @speakers_from_inline
    ) INTO @structured_json
    LOGGING 'Action items extracted' LEVEL INFO

    -- Step 5: Normalise relative due dates to ISO-8601 (deterministic — no LLM)
    CALL normalize_dates(@structured_json) INTO @structured_json

    -- Step 6: Validate ownership — flag unowned high-priority items (deterministic — no LLM)
    CALL validate_ownership(@structured_json) INTO @validation_notes

    -- Step 7: Format and commit output
    EVALUATE @output_format
        WHEN = 'markdown' THEN
            GENERATE format_as_markdown(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'markdown'
        WHEN = 'email' THEN
            GENERATE format_as_email(@structured_json, @validation_notes) INTO @output
            COMMIT @output WITH status = 'complete', format = 'email'
        ELSE
            SET @output = @structured_json
            COMMIT @output WITH status = 'complete', format = 'json'
    END

EXCEPTION
    WHEN ContextLengthExceeded THEN
        -- Long transcript: summarise first, then extract from the summary
        GENERATE summarize_transcript(@clean_transcript) INTO @summary
        GENERATE extract_actions(@summary, action_item_schema(), @speakers_from_file, @speakers_from_inline) INTO @structured_json
        CALL normalize_dates(@structured_json) INTO @structured_json
        CALL validate_ownership(@structured_json) INTO @validation_notes
        SET @output = @structured_json
        COMMIT @output WITH status = 'complete_chunked'
END

## Normalizing Conversations and Extracting Actions

Now, let's modify the `normalize_transcript` function to extract actions:

### Code

```python
import re
from collections import defaultdict, deque
import spacy

# Initialize spaCy model for NLP tasks
nlp = spacy.load("en_core_web_sm")

def normalize_transcript(transcripts):
    conversation_map = defaultdict(list)
    
    for transcript in transcripts:
        # Split transcript into lines
        lines = transcript.split('\n')
        
        # Extract speakers and text from each line
        speakers = [line.split(':')[0].strip() for line in lines]
        text = [line.split(':')[1].strip() for line in lines]
        
        # Map speaker to conversation ID
        conversation_id = speakers[0]
        
        # Add conversation data to map
        conversation_map[conversation_id].append((speakers, text))
    
    # Normalize the conversations and return them as a single string
    normalized_conversation = ''
    for conversation in conversation_map.values():
        if len(conversation) > 1:
            speakers, text = zip(*conversation)
            speaker_text_pairs = list(zip(speakers, text))
            
            # Sort the pairs based on speaker name
            sorted_pairs = sorted(speaker_text_pairs, key=lambda x: x[0])
            
            # Extract the conversation between two consecutive speakers
            for i in range(1, len(sorted_pairs)):
                normalized_conversation += f"{sorted_pairs[i-1][0]}: {sorted_pairs[i][1]}\n"
    
    # Extract actions from the conversation
    action_text = []
    for pair in normalized_conversation.split('\n'):
        if ':' in pair:
            speaker, message = pair.split(':', 1)
            doc = nlp(message.lower())  # Convert to lowercase for better entity recognition
            actions = [entity.text for entity in doc.ents if entity.label_ == 'ACTION']
            action_text.append(actions[0] if actions else '')
    
    return normalized_conversation, '\n'.join(action_text)

# Test the function with given inputs

]

normalized_conversation, actions = normalize_transcript(transcripts)
print(normalized_conversation)

### Explanation

1.  We initialize a spaCy model for NLP tasks.
2.  In the `normalize_transcript` function, we extract the conversation text and split it into individual lines.
3.  We then sort these pairs based on speaker name and join them together to form the normalized conversation string.
4.  Finally, we iterate over each line in the normalized conversation and use spaCy to identify any entities (actions) within that line.
5.  Note that we convert the message text to lowercase before processing it with spaCy to improve entity recognition.
