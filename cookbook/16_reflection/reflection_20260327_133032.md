-- Recipe 16: Reflection Agent
-- The agent solves a problem, then reflects on its own reasoning to catch errors.
-- If reflection finds issues, it corrects and re-reflects until confident.
-- Demonstrates meta-cognitive loops and self-assessment patterns.
--
-- Usage:
--   spl run cookbook/16_reflection/reflection.spl --adapter ollama -m gemma3 problem="What are the trade-offs of microservices vs monoliths?"
--   spl run cookbook/16_reflection/reflection.spl --adapter ollama problem="Design a URL shortener system"
--   spl run cookbook/16_reflection/reflection.spl --adapter ollama problem="Explain why bubble sort is O(n^2)"

WORKFLOW reflection_agent
    INPUT:
        @problem TEXT,
        @max_reflections Number DEFAULT 3,
        @log_dir TEXT DEFAULT 'cookbook/16_reflection/logs'
    OUTPUT: @answer TEXT
DO
    @iteration := 0
    @confidence := 0

    LOGGING f'Reflection agent started | max_reflections={@max_reflections}' LEVEL INFO

    -- Initial attempt at solving the problem
    GENERATE solve(@problem) INTO @answer
    LOGGING 'Initial solution ready' LEVEL INFO
    CALL write_file(f'{@log_dir}/answer_0.md', @answer) INTO @_

    -- Reflection loop: assess and improve
    WHILE @iteration < @max_reflections DO
        LOGGING f'Reflection iteration {@iteration} ...' LEVEL DEBUG
        -- Reflect: identify errors, gaps, or weak reasoning
        GENERATE reflect(@problem, @answer) INTO @reflection
        CALL write_file(f'{@log_dir}/reflection_{@iteration}.md', @reflection) INTO @_

        -- Score confidence in the answer
        GENERATE confidence_score(@answer, @reflection) INTO @confidence
        LOGGING f'Confidence score: {@confidence} | iteration={@iteration}' LEVEL DEBUG

        EVALUATE @confidence
            WHEN > 0.85 THEN
                LOGGING f'Confident at iteration {@iteration} | score={@confidence}' LEVEL INFO
                CALL write_file(f'{@log_dir}/final.md', @answer) INTO @_
                COMMIT @answer WITH status = 'confident', confidence = @confidence, reflections = @iteration
            ELSE
                -- Extract specific issues to fix
                GENERATE extract_issues(@reflection) INTO @issues

                -- Correct the answer based on identified issues
                GENERATE correct(@answer, @issues, @problem) INTO @answer
                @iteration := @iteration + 1
                CALL write_file(f'{@log_dir}/answer_{@iteration}.md', @answer) INTO @_
        END
    END

    -- If loop exhausted, commit best effort with final confidence
    LOGGING f'Max reflections reached | confidence={@confidence}' LEVEL WARN
    CALL write_file(f'{@log_dir}/final.md', @answer) INTO @_
    COMMIT @answer WITH status = 'best_effort', confidence = @confidence, reflections = @iteration

EXCEPTION
    WHEN MaxIterationsReached THEN
        CALL write_file(f'{@log_dir}/final.md', @answer) INTO @_
        COMMIT @answer WITH status = 'max_reflections', confidence = @confidence
    WHEN HallucinationDetected THEN
        GENERATE solve(@problem) INTO @answer
        COMMIT @answer WITH status = 'restarted'
END

from flask import Flask, request, jsonify
from flask_limiter import Limiter
from flask_limiter.util import get_remote_address

import redis  # Import for Redis rate limiting

app = Flask(__name__)
app.config["SECRET_KEY"] = "your_secret_key"  # Replace with a secret key

# Use Argon2 hash function as the unique identifier generator
def generate_unique_id(url):
    return hashlib.pbkdf2_hmac(
        'sha256', url.encode('utf-8'), os.urandom(16), 100000
    )

limiter = Limiter(key_func=get_remote_address, store=False)

@app.route('/shorten', methods=['POST'])
@limiter.limit("10/minute")
def shorten():
    try:
        # Validate the URL using urlparse
        from urllib.parse import urlparse
        parsed_url = urlparse(request.json['url'])

        if not all([parsed_url.scheme, parsed_url.netloc]):
            return jsonify({'error': 'Invalid URL'}), 400

        unique_id = generate_unique_id(parsed_url.geturl())
        shortened_url = f"http://short.url/{unique_id[:6]}"

        # Rate limiting logic
        redis_client = redis.Redis(host='localhost', port=6379, db=0)
        redis_client.incr(f"requests:{parsed_url.netloc}")

        if redis_client.get(f"requests:{parsed_url.netloc}") > 10:
            return jsonify({'error': 'Rate limit exceeded'}), 429

        # Store the shortened URL in Redis
        redis_client.set(f"{shortened_url}", unique_id)

        return jsonify({'shortened_url': shortened_url})

    except Exception as e:
        return jsonify({'error': str(e)}), 500
