-- Input validation → safe generation → output validation → response.
-- Demonstrates safety-first patterns: content filtering, PII detection, and output sanitization.
-- Four gates before a response is committed.
--
-- Usage (from catalog):

--   spl run cookbook/18_guardrails/guardrails.spl --adapter ollama -m gemma3 \
--       --tools cookbook/18_guardrails/tools.py \
--       user_input="My SSN is 123-45-6789, help me file taxes"

WORKFLOW guardrails_pipeline
    INPUT:  @input_id   TEXT DEFAULT '',
            @user_input TEXT DEFAULT ''
    OUTPUT: @safe_response TEXT
    SECURITY: CLASSIFICATION: internal
DO
    -- Step 1: Load test input from catalog (deterministic — no LLM)
    -- If input_id is blank, @user_input is used directly in subsequent steps.
    CALL load_test_input(@input_id) INTO @test_context

    -- Gate 1a: Keyword pre-screen (deterministic — no LLM)
    -- Fast check for obvious harmful / off-topic patterns before spending tokens.
    CALL classify_input_keywords(@user_input) INTO @keyword_class

    EVALUATE @keyword_class
        WHEN STARTSWITH 'harmful' THEN
            COMMIT 'I cannot help with that request.' WITH
                status        = 'blocked_harmful',
                gate          = 'keyword_prescreen',
                keyword_match = @keyword_class
        WHEN STARTSWITH 'off_topic' THEN
            COMMIT 'That question is outside my scope. Please ask something relevant.' WITH
                status        = 'blocked_off_topic',
                gate          = 'keyword_prescreen',
                keyword_match = @keyword_class
        ELSE
            -- Continue to LLM classification
    END

    -- Gate 1b: LLM input content classification (nuanced cases keywords miss)
    GENERATE classify_input(@user_input, @test_context) INTO @input_class

    EVALUATE @input_class
        WHEN 'harmful' THEN
            COMMIT 'I cannot help with that request.' WITH
                status = 'blocked_harmful',
                gate   = 'llm_classifier'
        WHEN 'off_topic' THEN
            COMMIT 'That question is outside my scope. Please ask something relevant.' WITH
                status = 'blocked_off_topic',
                gate   = 'llm_classifier'
        ELSE
            -- Input is acceptable, continue
    END

    -- Gate 2: PII detection (deterministic — no LLM)
    CALL detect_pii(@user_input) INTO @pii_report

    EVALUATE @pii_report
        WHEN STARTSWITH 'pii_found' THEN
            -- Gate 2b: Redact PII (deterministic — no LLM)
            CALL redact_pii(@user_input) INTO @clean_input
        ELSE
            SET @clean_input = @user_input
    END

    -- Gate 3: Generate response with safety-tuned prompt
    GENERATE safe_response(@clean_input) INTO @raw_response

    -- Gate 4: Output validation — check for PII leakage or harmful content
    GENERATE validate_output(@raw_response, @clean_input) INTO @output_check

    EVALUATE @output_check
        WHEN 'safe' THEN
            SET @safe_response = @raw_response
        WHEN 'contains_pii' THEN
            -- Deterministic PII redaction on the output too
            CALL redact_pii(@raw_response) INTO @safe_response
        WHEN 'hallucination' THEN
            GENERATE safe_response(@clean_input) INTO @safe_response
            RETRY WITH temperature = 0.1 LIMIT 3
        ELSE
            SET @safe_response = 'I was unable to generate a safe response. Please rephrase your question.'
    END

    COMMIT @safe_response WITH
        status       = 'complete',
        input_class  = @input_class,
        pii_detected = @pii_report

EXCEPTION
    WHEN RefusalToAnswer THEN
        COMMIT 'This request cannot be processed.' WITH status = 'refused'
    WHEN HallucinationDetected THEN
        COMMIT 'Unable to provide a verified response.' WITH status = 'hallucination_blocked'
    WHEN BudgetExceeded THEN
        COMMIT @safe_response WITH status = 'budget_limit'
END

class InputClassifier:
    def __init__(self):
        self.classifications = {
            "safe": ["safe_encryption", "safe_poem", "safe_recipe", "safe_explain_sql"],
            "pii": ["pii_ssn", "pii_credit_card", "pii_email", "pii_phone", "pii_multiple"],
            "harmful": ["harmful_malware", "harmful_violence", "harmful_phishing"],
            "off_topic": ["off_topic_stock", "off_topic_medical"],
            "edge": ["edge_pii_in_code", "edge_borderline", "edge_empty"]
        }

    def classify(self, input_str):
        if input_str in self.classifications["safe"]:
            return "Safe"
        elif input_str in self.classifications["pii"]:
            return "PII"
        elif input_str in self.classifications["harmful"]:
            return "Harmful"
        elif input_str in self.classifications["off_topic"]:
            return "Off Topic"
        elif input_str in self.classifications["edge"]:
            return "Edge"
        else:
            raise ValueError("Unknown input type")

classifier = InputClassifier()
print(classifier.classify("safe_encryption"))  # Output: Safe
print(classifier.classify("pii_ssn"))          # Output: PII
print(classifier.classify("harmful_malware"))   # Output: Harmful

This classifier uses a dictionary to map input types to their respective categories. The `classify` method takes an input string and returns the corresponding category. If the input is not recognized, it raises a `ValueError`.
