-- Recipe 19: Memory-Augmented Conversation
-- A conversational agent that remembers facts from prior turns using persistent memory.
-- Demonstrates STORE @var IN memory.<key>, context accumulation, and stateful interactions.
--
-- Usage:
--   spl run cookbook/19_memory_conversation/memory_chat.spl --adapter ollama 'user_input=My name is Alice and I am a data scientist'
--   spl run cookbook/19_memory_conversation/memory_chat.spl --adapter ollama 'user_input=What is my name?'
--   spl run cookbook/19_memory_conversation/memory_chat.spl --adapter ollama 'user_input=I prefer Python over R'
--   spl run cookbook/19_memory_conversation/memory_chat.spl --adapter ollama 'user_input=What do you know about me?'
--
-- Memory backend
-- ==============
-- Backend  : SQLite key-value store, managed automatically by the SPL runtime
-- Location : <cwd>/.spl/memory.db  (created on first write; path relative to where spl run is invoked)
-- Access   : memory.get('<key>') in SELECT, memory.<key> in STORE — no declaration needed
--
-- Keys used by this workflow:
--   chat_user_profile  — accumulated user facts as a bullet list (persists across runs)
--   chat_history       — recent conversation turns, trimmed to last 10 User/Assistant pairs
--
-- Memory management:
--   Inspect:  spl memory get chat_user_profile
--   Reset:    spl memory delete chat_user_profile && spl memory delete chat_history

-- Extract any personal facts the user shared. Return the exact string 'no_new_facts'
-- if nothing new was shared, otherwise return a bullet list of facts.
CREATE FUNCTION extract_facts(user_input TEXT) RETURNS TEXT AS $$
You are a fact extractor. Read the user message and extract any personal facts they shared
about themselves (name, job, preferences, location, etc.).

If no personal facts are present, reply with exactly: no_new_facts
Otherwise reply with a concise bullet list, one fact per line, e.g.:
- Name: Alice
- Role: data scientist
- Preference: Python over R

User message: "{user_input}"
$$;

-- Merge new facts into an existing profile. Return the updated profile as a bullet list.
CREATE FUNCTION merge_profile(existing_profile TEXT, new_facts TEXT) RETURNS TEXT AS $$
You are maintaining a user profile as a concise bullet list.
If the existing profile is empty or blank, start a new profile from the new facts only.
Otherwise merge the new facts in, keeping all prior facts unless directly contradicted.
Do NOT invent or assume any facts not explicitly stated.
Return ONLY the bullet list, no preamble.

Existing profile:
{existing_profile}

New facts to merge:
{new_facts}
$$;

-- Reply using full context: known profile + conversation history + current message.
CREATE FUNCTION contextual_reply(user_input TEXT, profile TEXT, history TEXT) RETURNS TEXT AS $$
You are a helpful assistant with memory. Use the user profile and conversation history
to give a personalised, contextually aware reply.

User profile:
{profile}

Recent conversation:
{history}

Current message: "{user_input}"
$$;

-- Trim conversation history to the last N turns.
CREATE FUNCTION trim_history(history TEXT, max_turns TEXT) RETURNS TEXT AS $$
You are a conversation history manager. Keep only the last {max_turns} User/Assistant
turn pairs from the history below. Preserve the original wording exactly.

History:
{history}
$$;

WORKFLOW memory_conversation
    INPUT: @user_input TEXT
    OUTPUT: @response TEXT
DO
    LOGGING f'Memory conversation | input: {@user_input}' LEVEL INFO

    -- Load existing user profile and chat history from .spl/memory.db (SQLite)
    SELECT memory.get('chat_user_profile') AS profile INTO @profile      -- returns "" if key absent
    SELECT memory.get('chat_history') AS chat_history INTO @chat_history -- returns "" if key absent
    LOGGING f'Profile loaded: {@profile}' LEVEL DEBUG

    -- Extract any new personal facts the user shared
    GENERATE extract_facts(@user_input) INTO @new_facts
    LOGGING f'New facts: {@new_facts}' LEVEL DEBUG

    -- Update profile only when new facts were actually found (exact string check)
    EVALUATE @new_facts
        WHEN = 'no_new_facts' THEN
            LOGGING 'No new facts — profile unchanged' LEVEL DEBUG
        ELSE
            LOGGING 'New facts detected — merging profile' LEVEL DEBUG
            GENERATE merge_profile(@profile, @new_facts) INTO @profile
            STORE @profile IN memory.chat_user_profile
    END

    -- Generate response with full context: profile + history + current input
    GENERATE contextual_reply(@user_input, @profile, @chat_history) INTO @response

    -- Persist updated chat history (keep last 10 turns)
    @chat_history := @chat_history || '\nUser: ' || @user_input || '\nAssistant: ' || @response
    GENERATE trim_history(@chat_history, 10) INTO @chat_history
    STORE @chat_history IN memory.chat_history

    LOGGING 'Response ready' LEVEL INFO
    COMMIT @response WITH status = 'complete'

EXCEPTION
    WHEN BudgetExceeded THEN
        COMMIT 'I remember you! But I ran out of budget for this response.' WITH status = 'budget_limit'
END
