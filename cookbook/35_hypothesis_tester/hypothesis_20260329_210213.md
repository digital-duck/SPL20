-- Recipe 35: Hypothesis Tester
-- Generate hypothesis → design test → evaluate evidence → conclude.
-- Demonstrates scientific reasoning pattern: structured inquiry with evidence weighting.
--
-- Usage:
--   spl run cookbook/35_hypothesis_tester/hypothesis.spl --adapter ollama -m gemma3 \
--       observation="Remote teams show lower productivity in the first month after joining"
--
--   spl run cookbook/35_hypothesis_tester/hypothesis.spl --adapter ollama \
--       observation="Users who receive onboarding emails churn 30% less" \
--       domain="product" confidence_threshold=0.75

CREATE FUNCTION hypothesis_framework()
RETURNS TEXT
AS $$
Scientific hypothesis format:
- Null hypothesis (H0): The effect does not exist / is due to chance
- Alternative hypothesis (H1): The proposed causal or correlational claim
- Key variables: independent, dependent, confounding
- Testability: How would we falsify H0?
$$;

CREATE FUNCTION evidence_schema()
RETURNS JSON
AS $$
{
  "type": "object",
  "properties": {
    "supports_h1":   {"type": "array", "items": {"type": "string"}},
    "supports_h0":   {"type": "array", "items": {"type": "string"}},
    "inconclusive":  {"type": "array", "items": {"type": "string"}},
    "confidence":    {"type": "number", "minimum": 0, "maximum": 1},
    "verdict":       {"type": "string", "enum": ["reject_h0", "fail_to_reject_h0", "inconclusive"]},
    "caveats":       {"type": "array", "items": {"type": "string"}}
  },
  "required": ["confidence", "verdict"]
}
$$;

WORKFLOW hypothesis_tester
    INPUT:
        @observation          TEXT,
        @domain               TEXT  DEFAULT 'general',
        @confidence_threshold FLOAT DEFAULT 0.7
    OUTPUT: @conclusion TEXT
DO
    LOGGING f'Hypothesis tester | domain={@domain} threshold={@confidence_threshold}' LEVEL INFO

    -- Step 1: Formulate hypotheses
    GENERATE formulate_hypotheses(
        @observation,
        @domain,
        hypothesis_framework()
    ) INTO @hypotheses
    LOGGING 'Hypotheses formulated' LEVEL DEBUG

    -- Step 2: Design a test plan
    GENERATE design_test(
        @hypotheses,
        @domain
    ) INTO @test_plan
    LOGGING 'Test plan designed' LEVEL DEBUG

    -- Step 3: Gather and evaluate available evidence
    GENERATE evaluate_evidence(
        @observation,
        @hypotheses,
        @test_plan,
        evidence_schema()
    ) INTO @evidence_json
    LOGGING 'Evidence evaluated' LEVEL DEBUG

    -- Step 4: Extract confidence score
    GENERATE extract_confidence(@evidence_json) INTO @confidence
    LOGGING f'Confidence score: {@confidence} | threshold={@confidence_threshold}' LEVEL INFO

    -- Step 5: Draw conclusion based on evidence strength
    EVALUATE @confidence
        WHEN >= @confidence_threshold THEN
            GENERATE write_conclusion(
                @hypotheses, @evidence_json, @test_plan, 'confident'
            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'concluded',
                verdict = 'h1_supported',
                confidence = @confidence
        WHEN >= 0.4 THEN
            GENERATE write_conclusion(
                @hypotheses, @evidence_json, @test_plan, 'uncertain'
            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'inconclusive',
                verdict = 'needs_more_data',
                confidence = @confidence
        ELSE
            GENERATE write_conclusion(
                @hypotheses, @evidence_json, @test_plan, 'refuted'
            ) INTO @conclusion
            COMMIT @conclusion WITH
                status = 'concluded',
                verdict = 'h0_not_rejected',
                confidence = @confidence
    END

EXCEPTION
    WHEN GenerationError THEN
        COMMIT @hypotheses WITH status = 'hypotheses_only', reason = 'evidence_evaluation_failed'
END

```python
import pandas as pd
import numpy as np

def evaluate_confidence(input_data):
    # Initialize an empty list to store hypotheses and their corresponding evidence
    hypotheses_with_evidence = []

    for index, hypothesis in input_data.iterrows():
        hypothesis_name = hypothesis['Hypothesis']
        measure = hypothesis['Measure']

        # Extract the confidence interval from the original DataFrame
        ci = input_data.loc[index, 'confidence_interval']

        # Calculate the lower and upper bounds of the CI
        if len(ci) == 2:
            lower_bound = np.percentile(input_data['measure'], ci[0])
            upper_bound = np.percentile(input_data['measure'], ci[1])
        elif len(ci) == 3:
            confidence = (ci[1] - ci[0]) / (len(input_data) * input_data['sample_size'].mean())
            z_score = 1.96
            lower_bound = measure - z_score * np.sqrt(measure) * np.sqrt(1/input_data['sample_size'])
            upper_bound = measure + z_score * np.sqrt(measure) * np.sqrt(1/input_data['sample_size'])

        # Create a new row with the hypothesis name, its measure and confidence interval
        row = {
            'Hypothesis': hypothesis_name,
            'Measure': measure,
            'Lower Bound': lower_bound,
            'Upper Bound': upper_bound,
            'Confidence Interval': f"{np.round(lower_bound*100,2)}% - {np.round(upper_bound*100,2)}%"
        }

        # Append the new row to the list of hypotheses with evidence
        hypotheses_with_evidence.append(row)

    return pd.DataFrame(hypotheses_with_evidence)
```
